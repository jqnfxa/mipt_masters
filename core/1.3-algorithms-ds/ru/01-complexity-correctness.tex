\section{Анализ алгоритмов: сложность и корректность}
\label{sec:1.3.01-complexity-correctness}

\begin{ticket}
Анализ алгоритмов. Понятие о сложности алгоритмов по времени и по памяти. Асимптотика, O-символика. Доказательство корректности алгоритмов.
\end{ticket}

\subsection*{Теория}

\emph{Алгоритмом} называется конечная детерминированная процедура,
которая по входу из заданной области выдаёт результат за конечное
число элементарных шагов. Изучение алгоритма имеет две стороны:
\emph{корректность} (всегда ли он даёт правильный ответ?) и
\emph{сложность} (насколько быстро он работает и сколько памяти
расходует?). Стандартная ссылка по обоим вопросам~--- \cite{CLRS2013}.

\subsubsection*{Модель стоимости}

Будем пользоваться моделью \emph{машины с произвольным доступом
(Random-Access Machine, RAM)}: каждая арифметическая операция,
сравнение, чтение и запись в ячейку памяти занимают одну единицу
времени, а в каждой ячейке хранится целое число фиксированного
размера, не зависящего от входа. \emph{Временем работы} алгоритма на
конкретном входе называется число выполненных элементарных операций.

\begin{definition}[Сложность по времени и по памяти]
\label{def:complexity}
Пусть $A$~--- алгоритм, $\mathcal{I}_n$~--- множество допустимых
входов \emph{размера}~$n$ (мера зависит от задачи: обычно это число
элементов на входе или число бит).
\begin{itemize}
  \item \emph{Сложность в худшем случае по времени}~---
        $T_A(n) = \max_{x \in \mathcal{I}_n} t_A(x)$,
        где $t_A(x)$~--- число элементарных операций при работе на
        входе~$x$.
  \item \emph{Сложность в лучшем случае}~---
        $\min_{x \in \mathcal{I}_n} t_A(x)$.
  \item \emph{Сложность в среднем} (относительно распределения
        $\mu_n$ на $\mathcal{I}_n$)~---
        $\E_{x \sim \mu_n}\, t_A(x)$.
  \item \emph{Сложность по памяти} $S_A(n)$~--- максимальное число
        ячеек памяти, используемых~$A$ на входах размера~$n$, не
        считая ячеек, в которых хранится сам вход.
\end{itemize}
Если не оговорено противное, под <<сложностью>> понимается сложность
\emph{в худшем случае}.
\end{definition}

\subsubsection*{Асимптотическая нотация}

Асимптотическая нотация скрывает константные множители и младшие
члены, фокусируя внимание на скорости роста при $n \to \infty$.

\begin{definition}[Большое-$O$, $\Omega$, $\Theta$, малое-$o$, малое-$\omega$]
\label{def:asymptotic}
Пусть $f, g \colon \N \to [0, \infty)$. Полагаем
\begin{align*}
  f(n) &= O(g(n))      &&\Leftrightarrow&& \exists\, C > 0,\, n_0 \in \N \;\;\forall n \geq n_0\colon\; f(n) \leq C\, g(n);\\
  f(n) &= \Omega(g(n)) &&\Leftrightarrow&& \exists\, c > 0,\, n_0 \in \N \;\;\forall n \geq n_0\colon\; f(n) \geq c\, g(n);\\
  f(n) &= \Theta(g(n)) &&\Leftrightarrow&& f(n) = O(g(n)) \text{ и } f(n) = \Omega(g(n));\\
  f(n) &= o(g(n))      &&\Leftrightarrow&& \forall\, \varepsilon > 0\;\; \exists\, n_0\;\;\forall n \geq n_0\colon\; f(n) \leq \varepsilon\, g(n);\\
  f(n) &= \omega(g(n)) &&\Leftrightarrow&& \forall\, K > 0\;\; \exists\, n_0\;\;\forall n \geq n_0\colon\; f(n) \geq K\, g(n).
\end{align*}
\end{definition}

\begin{remark}
Запись $f(n) = O(g(n))$~--- условность: знак <<$=$>> здесь обозначает
принадлежность классу $O(g)$ всех функций, мажорируемых~$g$ с точностью
до константы. В частности, отношение несимметрично:
$O(n) = O(n^2)$ читается слева направо.
\end{remark}

\begin{proposition}[Свойства $O$]
\label{prop:big-o-rules}
Для неотрицательных $f, f_1, f_2, g, g_1, g_2$ и $c > 0$:
\begin{enumerate}[label=(\alph*)]
  \item \emph{поглощение константы:} $c \cdot f(n) = O(f(n))$;
  \item \emph{транзитивность:} если $f = O(g)$ и $g = O(h)$, то $f = O(h)$;
  \item \emph{сумма:} если $f_1 = O(g_1)$ и $f_2 = O(g_2)$, то $f_1 + f_2 = O(\max\{g_1, g_2\}) = O(g_1 + g_2)$;
  \item \emph{произведение:} если $f_1 = O(g_1)$ и $f_2 = O(g_2)$, то $f_1 \cdot f_2 = O(g_1 \cdot g_2)$;
  \item \emph{логарифмы и степени:} для любых $a, b > 0$ выполнено
        $\log_a n = \Theta(\log_b n)$, и
        $\log^k n = o(n^\varepsilon)$ при любых $k > 0$ и $\varepsilon > 0$.
\end{enumerate}
\end{proposition}
\proofof{big-o-rules}

\subsubsection*{Основная теорема о рекуррентностях (формулировка)}

Многие алгоритмы <<разделяй и властвуй>> имеют время работы,
удовлетворяющее рекуррентности вида $T(n) = a\, T(n/b) + f(n)$ при
$a \geq 1$, $b > 1$. Следующее утверждение даёт асимптотику~$T$ в трёх
режимах.

\begin{theorem}[Основная теорема о рекуррентностях]
\label{thm:master}
Пусть $a \geq 1$, $b > 1$~--- константы и $f \colon \N \to [0, \infty)$.
Допустим, $T(n) = a\, T(n/b) + f(n)$ при $n \geq 1$ (с $T(1) = \Theta(1)$
и $n/b$, понимаемым как $\lfloor n/b \rfloor$ или $\lceil n/b \rceil$).
Положим $\alpha = \log_b a$. Тогда
\begin{enumerate}[label=(\alph*)]
  \item если $f(n) = O(n^{\alpha - \varepsilon})$ при некотором $\varepsilon > 0$,
        то $T(n) = \Theta(n^\alpha)$;
  \item если $f(n) = \Theta(n^\alpha)$,
        то $T(n) = \Theta(n^\alpha \log n)$;
  \item если $f(n) = \Omega(n^{\alpha + \varepsilon})$ при некотором $\varepsilon > 0$
        и $a\, f(n/b) \leq c\, f(n)$ при некотором $c < 1$ и всех
        достаточно больших~$n$, то $T(n) = \Theta(f(n))$.
\end{enumerate}
\end{theorem}

Доказательство \cref{thm:master} см.\ в~\cite[гл.~4]{CLRS2013}; в
дальнейшем используем теорему как готовое утверждение.

\subsubsection*{Корректность через инварианты циклов}

Стандартный приём строгого доказательства частичной корректности
итеративного алгоритма~--- \emph{инвариант цикла}: свойство~$I$,
которое выполнено перед началом тела цикла, сохраняется на каждой
итерации и из которого при завершении цикла вытекает постусловие.

\begin{theorem}[Принцип инварианта цикла]
\label{thm:loop-invariant}
Пусть у цикла есть условие на входе~$P$ (предусловие), условие
завершения~$\neg B$ и тело~$S$. Допустим, свойство~$I$ обладает
следующими тремя свойствами:
\begin{enumerate}[label=(\alph*)]
  \item \emph{инициализация:} $P \Rightarrow I$ выполнено перед первой
        итерацией;
  \item \emph{сохранение:} если $I \wedge B$ выполнено перед итерацией
        тела~$S$, то $I$ выполнено и после неё;
  \item \emph{завершение:} цикл рано или поздно завершается (то есть
        $\neg B$ становится истинным после конечного числа итераций).
\end{enumerate}
Тогда на выходе из цикла выполнено $I \wedge \neg B$. Если требуемое
постусловие~$Q$ следует из $I \wedge \neg B$, алгоритм
\emph{частично корректен}; вместе с пунктом~(в) он
\emph{полностью (тотально) корректен}.
\end{theorem}
\proofof{loop-invariant}

Для доказательства завершения обычно указывают \emph{вариант цикла}:
неотрицательное целочисленное выражение~$V$, строго убывающее на
каждой итерации. Поскольку строго убывающая последовательность в~$\N$
конечна, цикл обязательно завершается.

\subsection*{Доказательства}

\begin{proof}[\Cref{prop:big-o-rules}]
\proofanchor{big-o-rules}
\emph{(а)} При константе $C = c$ и $n_0 = 1$ имеем
$c f(n) \leq C f(n)$ тривиально.

\emph{(б)} Если $f(n) \leq C_1 g(n)$ при $n \geq n_1$ и
$g(n) \leq C_2 h(n)$ при $n \geq n_2$, то при
$n \geq \max\{n_1, n_2\}$ получаем $f(n) \leq C_1 C_2 h(n)$.

\emph{(в)} При $n$, превосходящих оба пороговых значения,
$f_1(n) + f_2(n) \leq C_1 g_1(n) + C_2 g_2(n) \leq (C_1 + C_2) \max\{g_1(n), g_2(n)\}$.
Эквивалентность с $O(g_1 + g_2)$ следует из
$\max\{g_1, g_2\} \leq g_1 + g_2 \leq 2 \max\{g_1, g_2\}$ для
неотрицательных аргументов.

\emph{(г)} При $n$, превосходящих оба пороговых значения,
$f_1(n) f_2(n) \leq C_1 C_2 g_1(n) g_2(n)$.

\emph{(д)} Замена основания: $\log_a n = \log_b n / \log_b a$~---
константный множитель, поэтому функции $\Theta$-эквивалентны. Для
$\log^k n = o(n^\varepsilon)$ возьмём логарифм и заметим, что
$\lim_{n \to \infty} (k \log \log n) / (\varepsilon \log n) = 0$,
поэтому $k \log \log n - \varepsilon \log n \to -\infty$, а значит,
$\log^k n / n^\varepsilon = e^{k \log\log n - \varepsilon \log n} \to 0$.
\end{proof}

\begin{proof}[\Cref{thm:loop-invariant}]
\proofanchor{loop-invariant}
Покажем индукцией по индексу итерации $k \geq 0$, что~$I$ выполнено
перед началом каждой итерации. База $k = 0$~--- это пункт~(а). Для
шага: если~$I$ выполнено перед итерацией~$k$ и тело цикла исполнилось
(то есть~$B$ тоже было истинным), то по~(б) после итерации~$I$
по-прежнему выполнено~--- это и есть начало итерации $k + 1$.

В силу~(в) цикл завершается на некоторой итерации $k^* < \infty$. На
выходе перед последней проверкой выполнялось~$I$, а проверка дала
$\neg B$, поэтому $I \wedge \neg B$ истинно. Заключение о частичной
или полной корректности вытекает из определений.
\end{proof}

\subsection*{Примеры}

\begin{example}[Линейный поиск]
\label{ex:linear-search}
Дан массив $A[1..n]$ и значение~$v$; просматривая массив слева
направо, возвращаем первый индекс~$i$ с $A[i] = v$ или
<<не найдено>>, если такого нет.

\emph{Худший случай.} Значения нет в массиве либо оно стоит на
последней позиции; цикл выполняет~$n$ сравнений, $T(n) = \Theta(n)$.

\emph{Лучший случай.} Значение в $A[1]$; одно сравнение, $\Theta(1)$.

\emph{Инвариант цикла.} <<В начале итерации с индексом~$i$ выполнено
$v \neq A[j]$ для всех $j < i$.>> Инициализация ($i = 1$) тривиальна;
сохранение~--- проверка в теле; на выходе либо мы вернули~$i$ (тогда
$A[i] = v$ и инвариант гарантирует, что это первое такое вхождение),
либо $i = n + 1$ (и $v \neq A[j]$ при всех $j \leq n$, так что ответ
<<не найдено>> верен).
\end{example}

\begin{example}[Бинарный поиск и основная теорема]
\label{ex:binary-search}
В отсортированном массиве $A[1..n]$ бинарный поиск~$v$ сравнивает~$v$
со средним элементом и рекурсивно сужает поиск до одной из половин.
Время работы удовлетворяет $T(n) = T(n/2) + \Theta(1)$ при $n > 1$ и
$T(1) = \Theta(1)$. Применяем \cref{thm:master} при $a = 1$, $b = 2$,
так что $\alpha = \log_2 1 = 0$ и
$f(n) = \Theta(1) = \Theta(n^0) = \Theta(n^\alpha)$. Случай~(б)
основной теоремы даёт $T(n) = \Theta(\log n)$.
\end{example}

\begin{example}[Сортировка слиянием и основная теорема]
\label{ex:mergesort}
Сортировка слиянием делит массив из~$n$ элементов на две половины,
рекурсивно сортирует каждую и сливает отсортированные половины за
$\Theta(n)$. Время работы удовлетворяет
$T(n) = 2 T(n/2) + \Theta(n)$. Применяем \cref{thm:master} при
$a = 2$, $b = 2$, $\alpha = 1$, $f(n) = \Theta(n) = \Theta(n^\alpha)$.
Случай~(б) даёт $T(n) = \Theta(n \log n)$. Сложность по памяти~---
$\Theta(n)$ за счёт вспомогательного буфера слияния.
\end{example}

\begin{example}[Сортировка вставками: полное доказательство корректности]
\label{ex:insertion-sort}
Сортировка вставками обрабатывает $A[1..n]$ слева направо; на $j$-й
итерации она вставляет очередной элемент $A[j]$ в уже отсортированный
префикс $A[1..j-1]$, сдвигая большие элементы на одну позицию вправо.
Время работы в худшем случае ($A$ упорядочен по убыванию)~---
$\Theta(n^2)$.

\emph{Инвариант цикла.} <<В начале итерации с индексом~$j$ (для
$j = 2, 3, \dots, n + 1$) подмассив $A[1..j-1]$ есть отсортированная
перестановка исходного $A[1..j-1]$.>>

\emph{Инициализация} ($j = 2$): префикс $A[1..1]$ тривиально
отсортирован и совпадает сам с собой.

\emph{Сохранение.} Пусть инвариант выполнен на итерации~$j$. Тело
сдвигает каждый элемент $A[1..j-1]$, превосходящий сохранённое
значение $A[j]$, на одну позицию вправо, а в образовавшийся пробел
помещает $A[j]$. Каждый сдвиг работает в пределах исходного
$\{A[1], \dots, A[j]\}$, так что $A[1..j]$~--- перестановка
исходного; сдвиги сохраняют отсортированность меньших элементов, а
вставленный элемент попадает ровно между наибольшим из элементов
$\leq A[j]$ и наименьшим из элементов $> A[j]$. Значит, $A[1..j]$
отсортирован.

\emph{Завершение.} Вариант $V = n + 1 - j$ строго убывает на
единицу с каждой итерацией внешнего цикла; при $V = 0$ цикл
выходит при $j = n + 1$. По инварианту $A[1..n]$ к этому моменту~---
отсортированная перестановка исходного, что и есть постусловие.
\end{example}

\begin{problem}[Иерархия типичных скоростей роста]
Расположите перечисленные функции по возрастанию асимптотической
скорости (от самой медленной к самой быстрой), группируя
$\Theta$-эквивалентные:
$1$, $\log\log n$, $\log n$, $\sqrt{n}$, $n$, $n \log n$, $n^2$,
$n^{\log_2 3}$, $2^n$, $n!$. Для каждого строгого неравенства приведите
обоснование, опираясь на \cref{prop:big-o-rules} либо вычисляя
$\lim_{n \to \infty} f(n)/g(n)$ через правило Лопиталя или формулу
Стирлинга.
\end{problem}
