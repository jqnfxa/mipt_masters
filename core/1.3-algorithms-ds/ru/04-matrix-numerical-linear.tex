% Topic 1.3.04 — Matrix Multiplication and Numerical Linear Solvers.

\section{Матрицы и численное решение СЛАУ}
\label{sec:1.3.04-matrix-numerical-linear}

\begin{ticket}
Представление матриц и векторов. Алгоритмы умножения матриц и эффективные способы их реализации. Численные методы решения систем линейных уравнений.
\end{ticket}

\subsection*{Теория}

В этом билете объединяются вопрос о структурах данных (как разместить
матрицы и векторы в памяти, чтобы циклы работали быстро) и два
алгоритмических вопроса (умножение матриц и решение
$A \vect{x} = \vect{b}$). Подробное изложение~--- в
\cite[гл.~28]{CLRS2013}.

\subsubsection*{Представление}

Вектор $\vect{v} \in \R^n$ хранится в виде непрерывного массива из
$n$ значений с плавающей точкой; обращение к элементу~--- $\Theta(1)$.

Матрица $A \in \R^{m \times n}$ имеет два основных плотных способа
размещения:
\begin{itemize}
  \item \emph{построчное (row-major)}: $A[i][j]$ по смещению
        $i \cdot n + j$ (используется в C, по умолчанию в NumPy);
  \item \emph{постолбцовое (column-major)}: $A[i][j]$ по смещению
        $j \cdot m + i$ (Fortran, BLAS, MATLAB).
\end{itemize}
Выбор существенен для поведения кэша: тесный цикл по непрерывной оси
получает выгоду от пространственной локальности, а проход по оси с
шагом приводит в худшем случае к одному кэш-промаху на элемент.

Для разреженной матрицы с $\mathrm{nnz}(A)$ ненулевыми элементами
распространены форматы:
\begin{itemize}
  \item \emph{COO} (координатный): три массива длины $\mathrm{nnz}(A)$
        с тройками $(i, j, A_{ij})$ для каждого ненулевого элемента;
  \item \emph{CSR} (compressed sparse row): значения $A_{ij}$ и
        столбцовые индексы сгруппированы по строкам, плюс массив
        начал строк длины $m + 1$; обеспечивает доступ к строке за
        $\Theta(\mathrm{nnz}(A_{i,*}))$.
\end{itemize}

\subsubsection*{Умножение плотных матриц}

Даны $A \in \R^{n \times n}$, $B \in \R^{n \times n}$; требуется
вычислить $C = AB \in \R^{n \times n}$ по формуле
$C_{ij} = \sum_{k=1}^{n} A_{ik} B_{kj}$.

\begin{proposition}[Наивное умножение матриц]
\label{prop:naive-matmul}
Алгоритм с тремя вложенными циклами
\[
  \text{для } i, j, k \in \{1, \dots, n\}\colon\quad C_{ij} \mathrel{+}= A_{ik}\, B_{kj}
\]
вычисляет $C = AB$ за $\Theta(n^3)$ времени и использует
$\Theta(n^2)$ дополнительной памяти (саму матрицу~$C$).
\end{proposition}
\proofof{naive-matmul}

\begin{theorem}[Штрассен, 1969]
\label{thm:strassen}
Существует алгоритм умножения двух матриц $n \times n$ за
$T(n) = 7\, T(n/2) + \Theta(n^2)$ арифметических операций, то есть за
$T(n) = \Theta(n^{\log_2 7}) = \Theta(n^{2{,}807\ldots})$.
\end{theorem}
\proofof{strassen}

\begin{remark}[Блочное умножение и эффективность кэша]
Даже без изменения асимптотического показателя разбиение $A$ и $B$ на
блоки $b \times b$ (с~$b$, при котором блок помещается в кэш) даёт
постоянный множитель ускорения относительно наивных трёх циклов на
современном железе: каждый загруженный блок переиспользуется
$\Theta(n/b)$ раз до вытеснения. При $b = \Theta(\sqrt{M})$, где
$M$~--- размер кэша, число кэш-промахов снижается с $\Theta(n^3)$ до
$\Theta(n^3 / \sqrt{M})$.
\end{remark}

\subsubsection*{Прямые методы}

Для невырожденной $A \in \R^{n \times n}$ и $\vect{b} \in \R^n$
\emph{метод Гаусса} приводит~$A$ к верхнетреугольному виду прямым
ходом исключения, после чего \emph{обратный ход} решает треугольную
систему. Эквивалентно, исключение даёт разложение $P A = L U$, где
$L$~--- нижнетреугольная с единицами на диагонали, $U$~---
верхнетреугольная, а $P$~--- матрица перестановок, фиксирующая
переупорядочения строк при \emph{выборе главного элемента по
столбцу} (на каждом шаге берётся ведущий элемент с наибольшим модулем
в текущем столбце, чтобы ограничить рост ошибок).

\begin{theorem}[Стоимость LU с частичным пивотированием]
\label{thm:lu-cost}
Вычисление разложения $P A = L U$ для $A \in \R^{n \times n}$ требует
$\tfrac{2}{3} n^3 + O(n^2)$ операций над числами с плавающей точкой и
$O(n^2)$ дополнительной памяти. После получения разложения решение
$A \vect{x} = \vect{b}$ через $L \vect{y} = P \vect{b}$ и далее
$U \vect{x} = \vect{y}$ требует $2 n^2 + O(n)$ операций (два
треугольных решения).
\end{theorem}
\proofof{lu-cost}

Для симметрично положительно определённой матрицы существует
аналогичное \emph{разложение Холецкого} $A = L L^\top$, требующее вдвое
меньше операций и не нуждающееся в пивотировании.

\subsubsection*{Итерационные методы}

Для больших разреженных~$A$ прямые методы упираются в стоимость
$\Theta(n^3)$ плотного разложения. Итерационные методы строят
последовательность $\vect{x}^{(0)}, \vect{x}^{(1)}, \dots$, сходящуюся
к точному решению.

\emph{Стационарная итерация} строится по разложению $A = M - N$, в
котором $M$ легко обращается:
\[
  \vect{x}^{(k+1)} \;=\; M^{-1} \bigl(N \vect{x}^{(k)} + \vect{b}\bigr)
                      \;=\; T \vect{x}^{(k)} + \vect{c},
  \qquad T = M^{-1} N, \;\vect{c} = M^{-1} \vect{b}.
\]
Два классических выбора при разложении $A = D + L_A + U_A$, где
$D$~--- диагональ, $L_A$ и $U_A$~--- строго нижне- и
верхнетреугольные части:
\begin{itemize}
  \item \emph{Якоби:} $M = D$, $N = -(L_A + U_A)$, тогда
        $T_J = -D^{-1}(L_A + U_A)$;
  \item \emph{Гаусс--Зейдель:} $M = D + L_A$, $N = -U_A$, тогда
        $T_{GS} = -(D + L_A)^{-1} U_A$.
\end{itemize}

\begin{theorem}[Сходимость стационарных итераций]
\label{thm:stationary-convergence}
Итерация $\vect{x}^{(k+1)} = T \vect{x}^{(k)} + \vect{c}$ сходится к
единственной неподвижной точке
$\vect{x}^* = (I - T)^{-1} \vect{c}$ из любого начального вектора
$\vect{x}^{(0)}$ тогда и только тогда, когда \emph{спектральный
радиус} $\rho(T) = \max\{|\lambda| : \lambda \text{ собственное значение } T\}$
строго меньше~$1$. Асимптотическая скорость убывания ошибки на одном
шаге~--- $\rho(T)$.
\end{theorem}
\proofof{stationary-convergence}

Распространённое достаточное условие: если~$A$ строго диагонально
доминирующая ($|A_{ii}| > \sum_{j \neq i} |A_{ij}|$ при всех~$i$), то
сходятся как метод Якоби, так и метод Гаусса--Зейделя.

\subsection*{Доказательства}

\begin{proof}[\Cref{prop:naive-matmul}]
\proofanchor{naive-matmul}
Тело $C_{ij} \mathrel{+}= A_{ik} B_{kj}$ выполняется по разу для
каждой тройки $(i, j, k) \in \{1, \dots, n\}^3$, то есть $n^3$ раз;
каждое исполнение~--- $O(1)$. Обнуление матрицы $C$ стоит
$\Theta(n^2)$. Итого $\Theta(n^3)$.
\end{proof}

\begin{proof}[\Cref{thm:strassen}]
\proofanchor{strassen}
Разобьём $A$, $B$, $C$ на $2 \times 2$ блоки размера
$n/2 \times n/2$:
\[
  A = \begin{pmatrix} A_{11} & A_{12} \\ A_{21} & A_{22} \end{pmatrix},\quad
  B = \begin{pmatrix} B_{11} & B_{12} \\ B_{21} & B_{22} \end{pmatrix}.
\]
Наивный блочный алгоритм использует $8$ умножений матриц размера
$n/2$, что приводит к $T(n) = 8 T(n/2) + \Theta(n^2) = \Theta(n^3)$~---
выигрыша нет. Штрассен указывает явные формулы для $C_{ij}$,
использующие лишь \emph{семь} блочных произведений
$M_1, \dots, M_7$ подходящих сумм и разностей блоков $A_{ij}$ и $B_{ij}$,
например
\[
  M_1 = (A_{11} + A_{22})(B_{11} + B_{22}),\quad
  M_2 = (A_{21} + A_{22}) B_{11},\quad \dots
\]
Блоки $C_{ij}$ затем выражаются как линейные комбинации
$M_1, \dots, M_7$. Рекуррентность принимает вид
$T(n) = 7 T(n/2) + \Theta(n^2)$; по основной теореме
(\cref{thm:master}) при $a = 7, b = 2, \alpha = \log_2 7 \approx 2{,}807$
и $f(n) = \Theta(n^2) = O(n^{\alpha - \varepsilon})$ для любого
$\varepsilon < \alpha - 2$ случай~(а) даёт
$T(n) = \Theta(n^{\log_2 7})$.
\end{proof}

\begin{proof}[\Cref{thm:lu-cost}]
\proofanchor{lu-cost}
Исключение столбца~$k$ ($k = 1, \dots, n - 1$) требует $n - k$
множителей (элементы под диагональю в столбце~$k$ матрицы~$L$) и
обновления оставшейся подматрицы $(n - k) \times (n - k)$~--- по
одному умножению и одному вычитанию на элемент: $(n - k)^2$ операций.

Общее число операций прямого хода:
\[
  \sum_{k=1}^{n-1} \bigl[(n - k) + 2 (n - k)^2\bigr]
  \;=\; \tfrac{n(n - 1)}{2} + 2 \cdot \tfrac{(n - 1) n (2n - 1)}{6}
  \;=\; \tfrac{2}{3} n^3 + O(n^2).
\]
Треугольные решения: прямое решение $L \vect{y} = P \vect{b}$
использует $\sum_{i=1}^{n} (i - 1)$ умножений и столько же вычитаний,
то есть $n^2 + O(n)$; обратный ход $U \vect{x} = \vect{y}$~--- ещё
$n^2 + O(n)$. Итого $2 n^2 + O(n)$.
\end{proof}

\begin{proof}[\Cref{thm:stationary-convergence}]
\proofanchor{stationary-convergence}
Положим $\vect{e}^{(k)} = \vect{x}^{(k)} - \vect{x}^*$. Вычитая из
итерации уравнение неподвижной точки
$\vect{x}^* = T \vect{x}^* + \vect{c}$, получаем
$\vect{e}^{(k+1)} = T \vect{e}^{(k)}$, откуда
$\vect{e}^{(k)} = T^k \vect{e}^{(0)}$.

\emph{Достаточность.} Если $\rho(T) < 1$, то $\|T^k\| \to 0$ в любой
матричной норме (формула Гельфанда:
$\rho(T) = \lim_k \|T^k\|^{1/k}$, так что для произвольного
$r \in (\rho(T), 1)$ начиная с некоторого~$k$ имеем
$\|T^k\| \leq r^k \to 0$). Значит, $\vect{e}^{(k)} \to 0$ при любом
начальном $\vect{e}^{(0)}$.

\emph{Необходимость.} Если $\rho(T) \geq 1$, возьмём собственную
пару $T \vect{v} = \lambda \vect{v}$ с $|\lambda| \geq 1$ и стартовую
ошибку $\vect{e}^{(0)} = \vect{v}$: тогда
$\|\vect{e}^{(k)}\| = |\lambda|^k \|\vect{v}\|$, и она не стремится
к нулю.

Асимптотическая скорость $\rho(T)$ следует из
$\|T^k\|^{1/k} \to \rho(T)$.
\end{proof}

\subsection*{Примеры}

\begin{example}[Наивный алгоритм против Штрассена при $n = 1024$]
\label{ex:naive-vs-strassen}
Наивное умножение делает $1024^3 \approx 1{,}07 \times 10^9$
скалярных умножений. Алгоритм Штрассена, рекурсивно опускаясь до
малого базового случая~$n_0$, делает примерно
$1024^{\log_2 7} \approx 1024^{2{,}807} \approx 1{,}32 \times 10^8$
скалярных умножений~--- примерно в $8$ раз меньше.

Перекрёсток, после которого константные накладные расходы Штрассена
начинают окупаться, на реальных машинах обычно $n_0 \approx 50$--$100$
(семь блочных произведений сопровождаются $18$ блочными
сложениями/вычитаниями против $8$ произведений и $4$ сложений у
наивного алгоритма). Промышленные библиотеки BLAS при $n \leq n_0$
переходят на оптимизированные три цикла.
\end{example}

\begin{example}[LU-разложение матрицы $3 \times 3$]
\label{ex:lu-3x3}
Пусть
$A = \begin{pmatrix} 2 & 1 & 1 \\ 4 & 3 & 3 \\ 8 & 7 & 9 \end{pmatrix}$.
Без выбора главного элемента:
\begin{enumerate}[label=(\arabic*)]
  \item вычитаем из строки~2 строку~1, умноженную на $\tfrac{4}{2} = 2$;
        строка~2 становится $(0, 1, 1)$;
  \item вычитаем из строки~3 строку~1, умноженную на $\tfrac{8}{2} = 4$;
        строка~3 становится $(0, 3, 5)$;
  \item вычитаем из строки~3 строку~2 (теперь $(0, 1, 1)$), умноженную
        на $\tfrac{3}{1} = 3$; строка~3 становится $(0, 0, 2)$.
\end{enumerate}
Множители разложения~---
$L = \begin{pmatrix} 1 & 0 & 0 \\ 2 & 1 & 0 \\ 4 & 3 & 1 \end{pmatrix}$,
$U = \begin{pmatrix} 2 & 1 & 1 \\ 0 & 1 & 1 \\ 0 & 0 & 2 \end{pmatrix}$.
Прямая проверка даёт $LU = A$. Решение системы
$A \vect{x} = (4, 11, 29)\T$ прямым и обратным ходами даёт
$\vect{y} = (4, 3, 4)\T$ и $\vect{x} = (1, 1, 2)\T$.
\end{example}

\begin{example}[Метод Якоби на маленькой положительно определённой системе]
\label{ex:jacobi}
Пусть $A = \begin{pmatrix} 4 & 1 \\ 1 & 3 \end{pmatrix}$,
$\vect{b} = (5, 4)\T$, точное решение $\vect{x}^* = (1, 1)\T$. Матрица
итераций Якоби~---
$T_J = -D^{-1}(L_A + U_A) = -\begin{pmatrix} 1/4 & 0 \\ 0 & 1/3 \end{pmatrix}\begin{pmatrix} 0 & 1 \\ 1 & 0 \end{pmatrix} = \begin{pmatrix} 0 & -1/4 \\ -1/3 & 0 \end{pmatrix}$.
Её собственные значения удовлетворяют $\lambda^2 = 1/12$, поэтому
$\rho(T_J) = 1/\sqrt{12} \approx 0{,}289$, и итерация сходится,
сокращая ошибку вдвое примерно за $\log_2(1/\rho) \approx 1{,}79$
шагов. Стартуя из $\vect{x}^{(0)} = 0$, получаем
$(5/4, 4/3) \approx (1{,}25,\, 1{,}33)$,
$(5/4 - 4/12, 4/3 - 5/12) \approx (0{,}917,\, 0{,}917)$~--- видно
сближение с $(1, 1)$.
\end{example}

\begin{problem}[Кэш-ориентированное блокирование]
Допустим, кэш вмещает $M$ значений с плавающей точкой, а матрицы $A$,
$B$, $C$ хранятся построчно. Покажите, что наивное умножение двух
матриц $n \times n$ с тремя вложенными циклами при $n^2 > M$
порождает $\Theta(n^3)$ кэш-промахов (полезного переиспользования
строк нет). Затем покажите, что блочное умножение с размером блока
$b = \Theta(\sqrt{M})$ требует лишь $\Theta(n^3 / \sqrt{M})$
кэш-промахов и достигает нижней границы Хонга--Кунга для умножения
матриц.
\end{problem}
