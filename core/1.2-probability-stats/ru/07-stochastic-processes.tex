\section{Случайные процессы и конечномерные распределения}
\label{sec:1.2.07-stochastic-processes}

\begin{ticket}
Определение случайного процесса. Система конечномерных распределений случайного процесса, ее свойства. Выборочное пространство случайного процесса.
\end{ticket}

\subsection*{Теория}

Случайный процесс~--- вероятностный аналог функции: вместо одного случайного исхода мы наблюдаем целое индексированное семейство исходов. Множество индексов может быть дискретным (моменты времени, метки), непрерывным (вещественное время) или многомерным (пространственным); значения~--- вещественными, векторными, множественными. Основной объект, к которому всё сводится,~--- совместные распределения \emph{конечного} числа координат.

\begin{definition}[Случайный процесс]
\label{def:stochastic-process}
\emph{Случайным процессом} на вероятностном пространстве $(\Omega, \mathcal{F}, \Prob)$ с множеством индексов~$T$ называется семейство $X = (X_t)_{t \in T}$ случайных величин $X_t \colon \Omega \to \R$. При фиксированном исходе $\omega \in \Omega$ функция $t \mapsto X_t(\omega)$ называется \emph{реализацией} (\emph{траекторией}) процесса~$X$.
\end{definition}

При $T = \N$ или $T = \Z$ говорят о случайном процессе \emph{с дискретным временем}; при $T = [0, \infty)$ или~$T = \R$~--- \emph{с непрерывным временем}. По умолчанию рассматриваются вещественнозначные процессы; векторнозначные и множественнозначные получаются заменой $X_t$ на отображение в $\R^d$ или другое измеримое пространство.

\medskip
\noindent\textbf{Выборочное пространство.} Каждому исходу~$\omega$ соответствует функция $t \mapsto X_t(\omega)$ из функционального пространства~$\R^T$. Отождествляя~$\omega$ с его траекторией, в качестве \emph{выборочного пространства} (или \emph{пространства траекторий}) можно взять само~$\R^T$, наделённое произведением (\emph{цилиндрической}) $\sigma$-алгеброй $\mathcal{B}(\R^T) = \bigotimes_{t \in T} \mathcal{B}(\R)$~--- наименьшей $\sigma$-алгеброй, относительно которой измеримы все координатные проекции $\pi_t \colon (x_s)_{s \in T} \mapsto x_t$. При таком соглашении распределение всего процесса~--- единственная вероятностная мера на $(\R^T, \mathcal{B}(\R^T))$.

\medskip
\noindent\textbf{Конечномерные распределения.} Цилиндрическая $\sigma$-алгебра порождена \emph{цилиндрическими множествами} вида $\{(x_s) : (x_{t_1}, \dots, x_{t_n}) \in B\}$ для конечных подмножеств $\{t_1, \dots, t_n\} \subseteq T$ и борелевских $B \in \mathcal{B}(\R^n)$. Поэтому вероятностная мера на $(\R^T, \mathcal{B}(\R^T))$ полностью определяется совместными распределениями конечных подсемейств $(X_{t_1}, \dots, X_{t_n})$.

\begin{definition}[Конечномерные распределения]
\label{def:fdd}
\emph{Системой конечномерных распределений} процесса~$X$ называется семейство вероятностных мер
\[
  \mu_{t_1, \dots, t_n}(B) \;=\; \Prob\bigl((X_{t_1}, \dots, X_{t_n}) \in B\bigr), \qquad B \in \mathcal{B}(\R^n),
\]
проиндексированных конечными упорядоченными наборами $(t_1, \dots, t_n) \in T^n$ для всех $n \geq 1$.
\end{definition}

\begin{proposition}[Согласованность конечномерных распределений]
\label{prop:fdd-consistency}
Конечномерные распределения произвольного случайного процесса удовлетворяют двум \emph{условиям согласованности Колмогорова}:
\begin{enumerate}[label=(C\arabic*)]
  \item \emph{инвариантность относительно перестановок:} для любой перестановки~$\pi$ множества $\{1, \dots, n\}$ и любых борелевских $B_1, \dots, B_n \in \mathcal{B}(\R)$
        \[
          \mu_{t_{\pi(1)}, \dots, t_{\pi(n)}}(B_{\pi(1)} \times \cdots \times B_{\pi(n)}) = \mu_{t_1, \dots, t_n}(B_1 \times \cdots \times B_n);
        \]
  \item \emph{маргинализация:} для любого $B \in \mathcal{B}(\R^n)$
        \[
          \mu_{t_1, \dots, t_n, t_{n+1}}(B \times \R) = \mu_{t_1, \dots, t_n}(B).
        \]
\end{enumerate}
\end{proposition}
\proofof{fdd-consistency}

Обратное утверждение~--- что любое согласованное семейство мер возникает как конечномерные распределения некоторого процесса~--- составляет содержание теоремы Колмогорова о существовании.

\begin{theorem}[Теорема Колмогорова о существовании]
\label{thm:kolmogorov-extension}
Пусть~$T$~--- произвольное множество индексов и для каждого конечного набора $(t_1, \dots, t_n) \in T^n$ задана вероятностная мера $\mu_{t_1, \dots, t_n}$ на $\mathcal{B}(\R^n)$. Если семейство $\{\mu_{t_1, \dots, t_n}\}$ удовлетворяет условиям согласованности (C1)--(C2) из~\cref{prop:fdd-consistency}, то существуют вероятностное пространство $(\Omega, \mathcal{F}, \Prob)$ и случайный процесс $(X_t)_{t \in T}$ на нём, конечномерные распределения которого совпадают с заданным семейством.
\end{theorem}
\proofof{kolmogorov-extension}

\begin{remark}
Теорема о существовании строит процесс на пространстве траекторий $(\R^T, \mathcal{B}(\R^T))$. Это пространство достаточно велико, чтобы кодировать все конечномерные распределения, но \emph{мало} в другом смысле: события, зависящие от несчётного числа координат~--- например, $\{\sup_{t \in [0, 1]} X_t \leq c\}$ при $T = [0, 1]$,~--- могут не принадлежать $\mathcal{B}(\R^T)$. Чтобы корректно работать с такими событиями, обычно выбирают \emph{регулярную модификацию} процесса, например, с непрерывными траекториями (\cite[гл.~II,~\S~2]{Sevastyanov1982}).
\end{remark}

\medskip
\noindent\textbf{Эквивалентные процессы.} Два процесса~$X$ и~$Y$, заданные, возможно, на разных вероятностных пространствах, но на общем множестве индексов~$T$, называются \emph{эквивалентными} (или \emph{одинаково распределёнными}), если их системы конечномерных распределений совпадают. Теорема Колмогорова показывает, что конечномерные распределения~--- полный инвариант на уровне $\mathcal{B}(\R^T)$, хотя эквивалентные процессы могут различаться как функции от~$\omega$ (это так называемая задача о выборе модификации).

\subsection*{Доказательства}

\begin{proof}[\Cref{prop:fdd-consistency}]
\proofanchor{fdd-consistency}
Оба условия следуют из стандартных свойств совместных распределений случайных величин на одном вероятностном пространстве.

\emph{(C1)} При любой перестановке~$\pi$ множества событий
\[
  \{(X_{t_1}, \dots, X_{t_n}) \in B_1 \times \cdots \times B_n\}
  \quad \text{и} \quad
  \{(X_{t_{\pi(1)}}, \dots, X_{t_{\pi(n)}}) \in B_{\pi(1)} \times \cdots \times B_{\pi(n)}\}
\]
совпадают как подмножества~$\Omega$~--- оба сводятся к $\{X_{t_i} \in B_i \text{ для каждого } i\}$,~--- значит, их вероятности равны.

\emph{(C2)} Множество $\{(X_{t_1}, \dots, X_{t_n}, X_{t_{n+1}}) \in B \times \R\}$ совпадает с $\{(X_{t_1}, \dots, X_{t_n}) \in B\}$, так как условие ``$X_{t_{n+1}} \in \R$'' выполнено автоматически. Поэтому соответствующие вероятности равны.
\end{proof}

\begin{proof}[\Cref{thm:kolmogorov-extension}, набросок]
\proofanchor{kolmogorov-extension}
Пусть $\mathcal{C}$~--- алгебра \emph{цилиндрических множеств} в~$\R^T$: множеств вида $C = \{(x_s) : (x_{t_1}, \dots, x_{t_n}) \in B\}$ для какого-то конечного $F = \{t_1, \dots, t_n\} \subseteq T$ и $B \in \mathcal{B}(\R^n)$. Положим $\nu(C) = \mu_F(B)$. Согласованность~(C1)--(C2) делает определение корректным: разные представления одного и того же цилиндра отвечают перестановкам индексов или вложениям в больший конечный набор индексов, а оба этих преобразования сохраняют значение~$\nu$.

Функция $\nu \colon \mathcal{C} \to [0, 1]$ конечно аддитивна (опять же сведением к общему конечному набору индексов) и $\sigma$-аддитивна на~$\mathcal{C}$~--- последнее и есть нетривиальный шаг, устанавливаемый компактным аргументом, который сводится к теореме Тихонова в~$\R^F$ (по существу: убывающая последовательность цилиндров с ненулевой мерой имеет непустое пересечение). Применяя теорему Каратеодори о продолжении, получаем единственное $\sigma$-аддитивное продолжение~$\Prob$ на~$\mathcal{B}(\R^T)$, а определение $X_t(\omega) = \omega_t$ задаёт процесс с предписанными конечномерными распределениями. Подробное изложение~--- в~\cite[гл.~II, \S~3]{Shiryaev1989}.
\end{proof}

\subsection*{Примеры}

\begin{example}[Бернуллиевский процесс]
\label{ex:bernoulli-process}
Пусть $T = \N$ и $(X_n)$~--- независимые одинаково распределённые $\mathrm{Bernoulli}(p)$. Конечномерные распределения являются произведениями: для любого конечного $\{n_1, \dots, n_k\} \subseteq \N$ и $\varepsilon_i \in \{0, 1\}$
\[
  \Prob(X_{n_1} = \varepsilon_1, \dots, X_{n_k} = \varepsilon_k) = \prod_{i=1}^{k} p^{\varepsilon_i} (1 - p)^{1 - \varepsilon_i}.
\]
Оба условия согласованности выполнены непосредственно (правая часть инвариантна относительно перестановок с точностью до переупорядочивания, а маргинализация уменьшает число сомножителей). Теорема Колмогорова о продолжении (\cref{thm:kolmogorov-extension}) гарантирует существование вероятностного пространства, на котором процесс задан, но в данном примере достаточно положить $\Omega = \{0,1\}^{\N}$ с произведением $\sigma$-алгебр и произведением мер.
\end{example}

\begin{example}[Простое симметричное случайное блуждание]
\label{ex:simple-rw}
В обозначениях~$ (X_n) $ из предыдущего примера при $p = 1/2$ положим $S_0 = 0$ и $S_n = (2 X_1 - 1) + \cdots + (2 X_n - 1)$~--- сумму~$\pm 1$-шагов. Конечномерные распределения~$(S_n)_{n \geq 0}$ выражаются через распределения~$(X_n)$ конечным линейным отображением, и согласованность переносится. Условное распределение $S_{n+1} \mid S_n$ равно $\frac{1}{2}\delta_{S_n + 1} + \frac{1}{2}\delta_{S_n - 1}$~--- простейший пример марковской цепи с дискретным временем (\cref{sec:1.2.09-markov-chains}).
\end{example}

\begin{example}[Гауссовский процесс, заданный конечномерными распределениями]
\label{ex:gaussian-process}
Пусть $T \subseteq \R$, фиксированы функция $m \colon T \to \R$ и неотрицательно определённое ядро $K \colon T \times T \to \R$ (то есть для всякого конечного $\{t_1, \dots, t_n\}$ матрица $(K(t_i, t_j))_{i, j}$ симметрична и неотрицательно определена). Объявим $\mu_{t_1, \dots, t_n}$ многомерным нормальным распределением на~$\R^n$ с вектором средних $(m(t_1), \dots, m(t_n))$ и ковариационной матрицей $K(t_i, t_j)$. Условия согласованности следуют из того, что маргинальные распределения многомерного нормального закона снова нормальны (с наследованным средним и ковариацией). По теореме Колмогорова существует случайный процесс~--- \emph{гауссовский процесс}~--- с предписанной структурой среднего и ковариации. Самый знаменитый пример~--- \emph{винеровский процесс} (броуновское движение): $m \equiv 0$ и $K(s, t) = \min(s, t)$ на $T = [0, \infty)$.
\end{example}

\begin{problem}
\label{prob:fdd-marginal}
Пусть случайный процесс $(X_t)_{t \in [0, 1]}$ имеет конечномерные распределения $\mu_{t_1, \dots, t_n}$, являющиеся совместно нормальными с $\E X_t = 0$ и $\Cov(X_s, X_t) = e^{-|s - t|}$. Проверьте условия согласованности и найдите распределение $X_t$ для одного~$t$ и распределение пары $(X_s, X_t)$.
\end{problem}

\begin{proof}[Решение]
$\Cov(X_t, X_t) = e^{0} = 1$, поэтому $X_t \sim \mathcal{N}(0, 1)$ при каждом~$t$. Для произвольной пары $s \leq t$ величины $(X_s, X_t)$ совместно нормальны с нулевым средним и ковариационной матрицей $\begin{pmatrix} 1 & e^{-(t - s)} \\ e^{-(t - s)} & 1 \end{pmatrix}$~--- в частности, $\rho(X_s, X_t) = e^{-(t - s)}$ экспоненциально затухает с ростом промежутка времени. Согласованность по перестановкам: ковариационное ядро симметрично, поэтому перестановка индексов согласованно переставляет элементы вектора средних и матрицы ковариаций. Маргинализация: маргинальное распределение $(n+1)$-мерного гауссовского закона по первым~$n$ координатам~--- $n$-мерный гауссовский закон с соответствующим средним и ковариацией (стандартный факт о маргиналах гауссовских векторов). Поэтому теорема Колмогорова (\cref{thm:kolmogorov-extension}) гарантирует существование процесса; это так называемый \emph{процесс Орнштейна--Уленбека}~--- стационарный гауссовский процесс, возникающий при обсуждении среднеквадратичной эргодичности (\cref{sec:1.2.08-ergodicity}).
\end{proof}
