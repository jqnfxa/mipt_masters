\section{Марковские цепи. Уравнения Колмогорова–Чепмена}
\label{sec:1.2.09-markov-chains}

\begin{ticket}
Марковский случайный процесс. Дискретная марковская цепь. Переходные вероятности, уравнения Колмогорова – Чепмена. Однородные дискретные марковские цепи.
\end{ticket}

\subsection*{Теория}

Марковский процесс~--- это такой случайный процесс, в котором будущее зависит от прошлого только через настоящее. Это простейший и наиболее распространённый способ строить зависимые процессы из ``одношагового'' рецепта переходов.

\begin{definition}[Марковское свойство]
\label{def:markov-property}
Случайный процесс $(X_n)_{n \geq 0}$ с дискретным временем и значениями в счётном множестве состояний~$S$ обладает \emph{марковским свойством}, если при любом $n \geq 0$, любых $i_0, \dots, i_{n+1} \in S$ и условии, что событие, на которое мы условно обусловливаем, имеет положительную вероятность, выполнено
\[
  \Prob(X_{n+1} = i_{n+1} \mid X_n = i_n, \dots, X_0 = i_0) \;=\; \Prob(X_{n+1} = i_{n+1} \mid X_n = i_n).
\]
Процесс с этим свойством называется \emph{(дискретной) марковской цепью} на~$S$.
\end{definition}

\begin{definition}[Переходные вероятности; однородность]
\label{def:transition-probabilities}
\emph{Переходной вероятностью} из состояния~$i$ в состояние~$j$ на шаге~$n$ называется
\[
  p_{ij}^{(n,n+1)} = \Prob(X_{n+1} = j \mid X_n = i).
\]
Цепь называется \emph{однородной} (\emph{по времени}), если $p_{ij}^{(n,n+1)} = p_{ij}$ не зависит от~$n$. \emph{Переходной матрицей} называется $\mat{P} = (p_{ij})_{i, j \in S}$; она \emph{стохастична}, то есть $p_{ij} \geq 0$ и $\sum_j p_{ij} = 1$ для каждого~$i$.
\end{definition}

По правилу умножения вероятностей (\cref{prop:multiplication-rule}) совместное распределение $(X_0, X_1, \dots, X_n)$ для однородной цепи раскладывается как
\[
  \Prob(X_0 = i_0, X_1 = i_1, \dots, X_n = i_n) \;=\; \Prob(X_0 = i_0) \cdot p_{i_0 i_1} \cdot p_{i_1 i_2} \cdots p_{i_{n-1} i_n},
\]
поэтому начальное распределение~$\pi_0$ на~$S$ вместе с переходной матрицей~$\mat{P}$ задаёт все конечномерные распределения цепи (\cref{def:fdd}); по теореме Колмогорова о существовании (\cref{thm:kolmogorov-extension}) этого достаточно, чтобы построить цепь на пространстве траекторий~$S^{\N}$.

\begin{definition}[$n$-шаговые переходные вероятности]
\label{def:n-step-transition}
Для однородной марковской цепи \emph{$n$-шаговой переходной вероятностью} называется
\[
  p_{ij}^{(n)} \;=\; \Prob(X_n = j \mid X_0 = i),
\]
с соглашением $p_{ij}^{(0)} = \delta_{ij}$ (символ Кронекера). Матрица $\mat{P}^{(n)} = (p_{ij}^{(n)})_{i,j \in S}$ также стохастична.
\end{definition}

\begin{theorem}[Уравнения Колмогорова--Чепмена]
\label{thm:kolmogorov-chapman}
Для однородной марковской цепи и любых $m, n \geq 0$
\[
  p_{ij}^{(n + m)} \;=\; \sum_{k \in S} p_{ik}^{(n)}\, p_{kj}^{(m)} \qquad \text{для всех } i, j \in S,
\]
или, в матричной записи, $\mat{P}^{(n+m)} = \mat{P}^{(n)}\, \mat{P}^{(m)}$. В частности, $\mat{P}^{(n)} = \mat{P}^n$~--- матрица $n$-шаговых переходов является $n$-й степенью одношаговой переходной матрицы.
\end{theorem}
\proofof{kolmogorov-chapman}

\begin{corollary}[Распределение в момент~$n$]
\label{cor:distribution-at-n}
Если~$X_0$ имеет распределение $\pi_0 = (\pi_0(i))_{i \in S}$ (записанное вектор-строкой), то $X_n$ имеет распределение $\pi_n = \pi_0\, \mat{P}^n$.
\end{corollary}
\proofof{distribution-at-n}

\medskip
\noindent\textbf{Стационарное распределение.} Распределение~$\pi$ на~$S$ называется \emph{стационарным} (\emph{инвариантным}), если $\pi = \pi\, \mat{P}$, то есть $\pi(j) = \sum_i \pi(i) p_{ij}$ для каждого $j \in S$. По индукции $\pi = \pi\, \mat{P}^n$ при всех $n \geq 0$, так что цепь, запущенная из стационарного распределения, остаётся в нём при любом времени. Для конечных цепей существование стационарного распределения вытекает из теоремы Перрона--Фробениуса, применённой к $\mat{P}^{\T}$; единственность и сходимость $\pi_0 \mat{P}^n \to \pi$ требуют неприводимости и апериодичности~--- стандартная эргодическая теорема для марковских цепей~\cite[гл.~VIII]{Shiryaev1989}.

\begin{remark}
Марковские процессы с непрерывным временем~--- их переходный механизм задаётся не матрицей, а инфинитезимальным генератором~--- строятся аналогично. Простейший случай~--- задача о ``конкурирующих экспонентах'' (\cref{prob:exponential-min}): время пребывания в каждом состоянии распределено экспоненциально, а в момент перехода следующее состояние выбирается цепью на скачках. См.~\cite[гл.~VIII]{Sevastyanov1982}.
\end{remark}

\subsection*{Доказательства}

\begin{proof}[\Cref{thm:kolmogorov-chapman}]
\proofanchor{kolmogorov-chapman}
По формуле полной вероятности (\cref{thm:total-probability}), применённой к разбиению $\{X_n = k\}_{k \in S}$,
\[
  \Prob(X_{n+m} = j \mid X_0 = i) = \sum_{k \in S} \Prob(X_{n+m} = j,\, X_n = k \mid X_0 = i).
\]
По правилу умножения каждое слагаемое равно $\Prob(X_n = k \mid X_0 = i) \cdot \Prob(X_{n+m} = j \mid X_n = k, X_0 = i)$. Марковское свойство сжимает второй множитель: $\Prob(X_{n+m} = j \mid X_n = k, X_0 = i) = \Prob(X_{n+m} = j \mid X_n = k) = p_{kj}^{(m)}$ (последнее равенство опирается на однородность по времени, поскольку условное распределение цепи через~$m$ шагов от состояния~$k$ в любой момент времени одно и то же). Складывая,
\[
  p_{ij}^{(n+m)} = \sum_{k} p_{ik}^{(n)} p_{kj}^{(m)}, \qquad \mat{P}^{(n+m)} = \mat{P}^{(n)} \mat{P}^{(m)}.
\]
Полагая $\mat{P}^{(0)} = \mat{I}$ и итерируя из $\mat{P}^{(1)} = \mat{P}$, по индукции получаем $\mat{P}^{(n)} = \mat{P}^n$.
\end{proof}

\begin{proof}[\Cref{cor:distribution-at-n}]
\proofanchor{distribution-at-n}
По формуле полной вероятности с условиями на начальное состояние,
\[
  \pi_n(j) = \Prob(X_n = j) = \sum_{i \in S} \Prob(X_0 = i) \Prob(X_n = j \mid X_0 = i) = \sum_i \pi_0(i) p_{ij}^{(n)},
\]
а это $j$-я компонента произведения вектор-строки на матрицу $\pi_0 \mat{P}^{(n)} = \pi_0 \mat{P}^n$.
\end{proof}

\subsection*{Примеры}

\begin{example}[Двухсостояние: погода]
\label{ex:two-state-chain}
Пусть $S = \{R, S\}$ (дождливо, солнечно) и
\[
  \mat{P} = \begin{pmatrix} 0{,}7 & 0{,}3 \\ 0{,}4 & 0{,}6 \end{pmatrix},
\]
где строки индексируют текущее состояние, а столбцы~--- следующее. Чтобы найти вероятность дождя через два дня при условии, что сегодня солнечно, нужно прочитать $p_{SR}^{(2)} = (\mat{P}^2)_{SR}$:
\[
  \mat{P}^2 = \begin{pmatrix} 0{,}61 & 0{,}39 \\ 0{,}52 & 0{,}48 \end{pmatrix},
\]
откуда $p_{SR}^{(2)} = 0{,}52$. Единственное стационарное распределение находится из системы $\pi = \pi \mat{P}$ с условием $\pi(R) + \pi(S) = 1$: $0{,}3\, \pi(R) = 0{,}4\, \pi(S)$, откуда $\pi(R) = 4/7$, $\pi(S) = 3/7$~--- асимптотическая частота дождливых дней.
\end{example}

\begin{example}[Простое случайное блуждание на~$\Z$]
\label{ex:rw-on-Z}
Положим $X_{n+1} = X_n + \xi_{n+1}$, где~$\xi_n$ независимы и одинаково распределены: $+1$ с вероятностью~$p$ и $-1$ с вероятностью~$1 - p$. Множество состояний~--- $\Z$, цепь однородна, переходные вероятности
\[
  p_{i, i+1} = p, \qquad p_{i, i-1} = 1 - p, \qquad p_{i, j} = 0 \text{ для остальных } j.
\]
Двухшаговые переходы: $p_{i, i}^{(2)} = 2 p (1 - p)$, $p_{i, i+2}^{(2)} = p^2$, $p_{i, i-2}^{(2)} = (1-p)^2$. Случайное блуждание посещает любое заданное состояние с вероятностью~$1$ при $p = 1/2$ (возвратность) и с вероятностью строго меньше~$1$ при $p \neq 1/2$ (невозвратность); чистейшее доказательство использует производящие функции~\cite[гл.~XII]{Sevastyanov1982}.
\end{example}

\begin{example}[Урна Эренфеста]
\label{ex:ehrenfest}
$N$~шаров распределены между двумя урнами. На каждом шаге равновероятно выбирается один шар и перекладывается в другую урну. Пусть $X_n$~--- число шаров в первой урне. Тогда $S = \{0, 1, \dots, N\}$ и
\[
  p_{i, i-1} = \frac{i}{N}, \qquad p_{i, i+1} = \frac{N - i}{N}, \qquad p_{i, j} = 0 \text{ для остальных } j.
\]
Прямая проверка показывает, что биномиальное распределение $\pi(i) = \binom{N}{i} 2^{-N}$ стационарно. Модель Эренфеста наглядно показывает, как ``квазидетерминированное'' больцмановское ``равновесие'' возникает из перемешивающей марковской цепи на конечном множестве состояний.
\end{example}

\begin{problem}
\label{prob:absorbing-chain}
Игрок с капиталом $i \in \{0, 1, \dots, N\}$ выигрывает или проигрывает по~$1$ единице за раунд независимо с вероятностью $1/2$ каждый исход; игра заканчивается при достижении~$0$ (разорение) или~$N$ (джекпот). Найдите вероятность $h_i = \Prob(\text{достичь } N \text{ раньше } 0 \mid X_0 = i)$.
\end{problem}

\begin{proof}[Решение]
Состояния~$0$ и~$N$~--- поглощающие ($p_{00} = p_{NN} = 1$). Во внутренних состояниях марковское свойство даёт линейное рекуррентное соотношение $h_i = \tfrac{1}{2} h_{i+1} + \tfrac{1}{2} h_{i-1}$ при $1 \leq i \leq N - 1$, с граничными условиями $h_0 = 0$ и $h_N = 1$. Соотношение перепишется как $h_{i+1} - h_i = h_i - h_{i-1}$, то есть разности постоянны; обозначим их через~$c$. Тогда $h_N - h_0 = N c = 1$, откуда $c = 1/N$ и
\[
  h_i \;=\; \frac{i}{N}.
\]
При несимметричных вероятностях шага $p \neq 1/2$ то же условие даёт нетривиальное геометрическое решение $h_i = (1 - r^i)/(1 - r^N)$ с $r = (1 - p)/p$~--- классическая формула \emph{разорения игрока}~\cite[гл.~VI, \S~3]{Shiryaev1989}.
\end{proof}
