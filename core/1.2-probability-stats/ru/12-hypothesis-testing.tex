\section{Статистические гипотезы. Ошибки I и II рода}
\label{sec:1.2.12-hypothesis-testing}

\begin{ticket}
Выбор статистических гипотез. Простые и сложные статистические гипотезы. Статистическое решение и решающее правило. Рандомизированные решающие правила. Ошибки первого и второго рода, мощность статистического критерия. Наиболее мощный и равномерно наиболее мощный критерий.
\end{ticket}

\subsection*{Теория}

\emph{Статистическая гипотеза}~--- это утверждение о том, какой именно вероятностный закон на пространстве выборок породил наблюдаемые данные. Проверка гипотезы~--- это выбор между двумя конкурирующими утверждениями по данным с одновременным контролем частоты ошибочных решений.

\medskip
\noindent\textbf{Статистическая модель и гипотезы.}

\begin{definition}[Гипотеза]
\label{def:hypothesis}
\emph{Статистической моделью} называется семейство вероятностных мер $\{\Prob_\theta : \theta \in \Theta\}$ на пространстве выборок $(\mathcal{X}, \mathcal{A})$, индексированное параметром $\theta \in \Theta$. \emph{Парой гипотез} $(H_0, H_1)$ называется разбиение $\Theta = \Theta_0 \sqcup \Theta_1$ на \emph{нулевую} гипотезу $H_0 : \theta \in \Theta_0$ и \emph{альтернативу} $H_1 : \theta \in \Theta_1$. Гипотеза называется \emph{простой}, если её множество параметров одноточечно; в противном случае~--- \emph{сложной}.
\end{definition}

\medskip
\noindent\textbf{Критерии, ошибки и мощность.}

\begin{definition}[Решающее правило]
\label{def:decision-rule}
\emph{(Нерандомизированным) решающим правилом} (\emph{критерием}) называется измеримая функция $\varphi \colon \mathcal{X} \to \{0, 1\}$, в которой $\varphi(x) = 1$ интерпретируется как ``отвергнуть~$H_0$'', а $\varphi(x) = 0$~--- как ``не отвергать''. \emph{Критическим множеством} называется $\mathcal{R} = \{x \in \mathcal{X} : \varphi(x) = 1\}$. \emph{Рандомизированное} решающее правило отличается тем, что значения~$\varphi$ берутся из отрезка~$[0, 1]$: $\varphi(x)$ задаёт условную вероятность отвергнуть~$H_0$ при наблюдении~$x$.
\end{definition}

\begin{definition}[Ошибки и мощность]
\label{def:errors-power}
Для (возможно, рандомизированного) критерия~$\varphi$ в указанной модели \emph{функция мощности}~---
\[
  \beta_\varphi(\theta) \;=\; \E_\theta\, \varphi \;=\; \Prob_\theta(\text{отвергнуть } H_0).
\]
\begin{itemize}
  \item Для $\theta \in \Theta_0$ величина $\beta_\varphi(\theta)$~--- вероятность \emph{ошибки I рода} (отвергнуть верную нулевую гипотезу).
  \item Для $\theta \in \Theta_1$ \emph{вероятность ошибки II рода} равна $1 - \beta_\varphi(\theta) = \Prob_\theta(\text{не отвергнуть})$, а \emph{мощность} критерия~$\varphi$ в точке~$\theta$ совпадает с самой $\beta_\varphi(\theta)$.
\end{itemize}
\emph{Размером} (\emph{уровнем значимости}) критерия называется $\sup_{\theta \in \Theta_0} \beta_\varphi(\theta)$.
\end{definition}

\medskip
\noindent\textbf{Наиболее мощные и равномерно наиболее мощные критерии.}

\begin{definition}[(Равномерно) наиболее мощный критерий]
\label{def:ump}
Критерий~$\varphi^*$ размера $\alpha \in (0, 1)$ называется \emph{наиболее мощным} (НМ) для проверки $H_0 : \theta \in \Theta_0$ против простой альтернативы $H_1 : \theta = \theta_1$, если для любого критерия~$\varphi$ размера не более~$\alpha$ выполнено $\beta_\varphi(\theta_1) \leq \beta_{\varphi^*}(\theta_1)$. Критерий называется \emph{равномерно наиболее мощным} (РНМ) размера~$\alpha$ для сложной альтернативы $H_1 : \theta \in \Theta_1$, если указанное неравенство выполнено для каждого $\theta_1 \in \Theta_1$ одновременно.
\end{definition}

\begin{theorem}[Лемма Неймана--Пирсона]
\label{thm:neyman-pearson}
Рассмотрим проверку простой нулевой гипотезы $H_0 : \theta = \theta_0$ против простой альтернативы $H_1 : \theta = \theta_1$ при плотностях $f_0$ и $f_1$ относительно общей доминирующей меры на~$\mathcal{X}$. При каждом $k \geq 0$ \emph{критерий отношения правдоподобия}
\[
  \varphi_k(x) \;=\; \begin{cases}
    1, & f_1(x) > k\, f_0(x), \\
    0, & f_1(x) < k\, f_0(x),
  \end{cases}
\]
(на множестве равенства $\{f_1 = k f_0\}$ значение~$\varphi_k$ выбирается так, чтобы добиться нужного размера $\alpha \in (0, 1)$) является наиболее мощным размера~$\alpha$ в смысле~\cref{def:ump}. Обратно, любой НМ-критерий размера~$\alpha$ совпадает с~$\varphi_k$ при некотором $k \geq 0$ почти наверное по~$\Prob_{\theta_0}$ и $\Prob_{\theta_1}$ вне множества равенства.
\end{theorem}
\proofof{neyman-pearson}

\begin{remark}
Лемма сводит задачу о наилучшем бинарном критерии к пороговой проверке отношения правдоподобия. Для сложной альтернативы РНМ-критерий существует в благоприятных случаях (например, в одно\-па\-ра\-мет\-ри\-чес\-ких экспоненциальных семействах с монотонным отношением правдоподобия), но не всегда; в гауссовской модели для двусторонней альтернативы стандартный ``симметричный'' критерий уже не РНМ, и приходится опираться на другие принципы~--- несмещённость, инвариантность, отношение правдоподобия против всей альтернативы~\cite[гл.~VII]{Shiryaev1989}.
\end{remark}

\medskip
\noindent\textbf{Стандартные $z$- и $t$-критерии.} Гауссовская теорема о выборочных среднем и дисперсии (\cref{thm:sample-mean-variance}) порождает два самых употребительных в классической статистике критерия.

\begin{proposition}[Критерии для среднего нормальной выборки]
\label{prop:z-and-t-tests}
Пусть $X_1, \dots, X_n$~--- независимы и одинаково распределены, $X_i \sim \mathcal{N}(\mu, \sigma^2)$.
\begin{enumerate}[label=(\alph*)]
  \item \emph{Известная~$\sigma$.} Для проверки $H_0 : \mu = \mu_0$ против $H_1 : \mu > \mu_0$ \emph{$z$-критерий}, отвергающий~$H_0$ при $z = \sqrt n (\bar X - \mu_0)/\sigma > z_\alpha$ (верхняя $\alpha$-квантиль распределения $\mathcal{N}(0, 1)$), является РНМ размера~$\alpha$.
  \item \emph{Неизвестная~$\sigma$.} Для тех же гипотез \emph{$t$-критерий}, отвергающий~$H_0$ при $t = \sqrt n (\bar X - \mu_0)/S > t_{\alpha, n - 1}$ (верхняя $\alpha$-квантиль распределения $t_{n - 1}$), имеет размер ровно~$\alpha$ при $H_0$ (статистика распределена по $t_{n - 1}$ согласно~\cref{thm:sample-mean-variance}).
\end{enumerate}
Двусторонние варианты отвергают $H_0$ при $|z| > z_{\alpha/2}$ или $|t| > t_{\alpha/2, n - 1}$.
\end{proposition}
\proofof{z-and-t-tests}

\subsection*{Доказательства}

\begin{proof}[\Cref{thm:neyman-pearson}, набросок]
\proofanchor{neyman-pearson}
Пусть $\varphi^* = \varphi_k$~--- критерий размера~$\alpha$, то есть $\E_{\theta_0} \varphi^* = \alpha$, и пусть~$\varphi$~--- произвольный критерий размера не более~$\alpha$. Рассмотрим интеграл $\int (\varphi^* - \varphi)(f_1 - k f_0)$ по~$\mathcal{X}$. По определению~$\varphi^*$ подынтегральное выражение поточечно неотрицательно:
\begin{itemize}
  \item на $\{f_1 > k f_0\}$ имеем $\varphi^* = 1 \geq \varphi$ и $f_1 - k f_0 > 0$;
  \item на $\{f_1 < k f_0\}$ имеем $\varphi^* = 0 \leq \varphi$ и $f_1 - k f_0 < 0$;
  \item на $\{f_1 = k f_0\}$ выполнено $f_1 - k f_0 = 0$.
\end{itemize}
Интегрируя, получаем
\[
  0 \leq \int (\varphi^* - \varphi)(f_1 - k f_0) = \bigl(\E_{\theta_1} \varphi^* - \E_{\theta_1} \varphi\bigr) - k \bigl(\E_{\theta_0} \varphi^* - \E_{\theta_0} \varphi\bigr).
\]
Из $\E_{\theta_0} \varphi^* = \alpha$ и $\E_{\theta_0} \varphi \leq \alpha$ следует, что вторая скобка не положительна, а потому $-k$ умноженное на неё неотрицательно. Значит, вся правая часть, оставаясь неотрицательной, требует $\E_{\theta_1} \varphi^* \geq \E_{\theta_1} \varphi$~--- мощность~$\varphi^*$ в точке~$\theta_1$ доминирует над мощностью любого другого критерия размера не более~$\alpha$. Обратное утверждение описывает структуру оптимума: равенство $\varphi = \varphi^*$ вне множества $\{f_1 = k f_0\}$ вытекает из строгих неравенств в первых двух пунктах, поскольку там подынтегральное выражение строго положительно, и для конкурирующего оптимума интеграл должен быть равен нулю.
\end{proof}

\begin{proof}[\Cref{prop:z-and-t-tests}]
\proofanchor{z-and-t-tests}
\emph{(a)} При известной~$\sigma$ совместная плотность выборки из $\mathcal{N}(\mu, \sigma^2)$ раскладывается как $f_\mu(\vect{x}) \propto \exp\bigl(\mu \sum_i x_i / \sigma^2 - n \mu^2 / (2 \sigma^2)\bigr)$ умножить на не зависящий от~$\mu$ множитель. Отсюда отношение правдоподобия $f_{\mu_1}(\vect{x})/f_{\mu_0}(\vect{x})$ при $\mu_1 > \mu_0$ является возрастающей функцией от~$\sum_i x_i$ (равно как и от~$\bar X$). По лемме Неймана--Пирсона (\cref{thm:neyman-pearson}) при любом конкретном $\mu_1 > \mu_0$ НМ-критерий отвергает~$H_0$ при больших значениях~$\bar X$; пороговое значение зависит лишь от размера~$\alpha$ и не от выбранного~$\mu_1$. Поэтому критерий ``отвергнуть при $\bar X > c_\alpha$'' является НМ для каждого $\mu_1 > \mu_0$ одновременно, то есть РНМ для сложной альтернативы $\mu > \mu_0$. Стандартизуя, $c_\alpha$ соответствует $z = \sqrt n (\bar X - \mu_0)/\sigma > z_\alpha$.

\emph{(b)} Утверждение о размере следует непосредственно из~\cref{thm:sample-mean-variance}: при $H_0 : \mu = \mu_0$ статистика $T = \sqrt n (\bar X - \mu_0)/S \sim t_{n - 1}$, и потому $\Prob_{\mu_0}(T > t_{\alpha, n - 1}) = \alpha$. (РНМ-свойство для $t$-критерия не выполнено в полном объёме из-за наличия мешающего параметра~$\sigma$ при альтернативе $\mu > \mu_0$; стандартный $t$-критерий является РНМ в более узких классах~--- среди \emph{несмещённых} или \emph{инвариантных} критериев~\cite[гл.~VII]{Shiryaev1989}.)
\end{proof}

\subsection*{Примеры}

\begin{example}[Расчёт мощности $z$-критерия]
\label{ex:z-test-power}
Проверим $H_0 : \mu = 0$ против $H_1 : \mu > 0$ по выборке размера $n = 25$ из $\mathcal{N}(\mu, 1)$ на уровне $\alpha = 0{,}05$. РНМ-критерий (\cref{prop:z-and-t-tests}\,(a)) отвергает~$H_0$ при $\bar X > z_{0{,}05}/\sqrt{25} = 1{,}645/5 = 0{,}329$. Мощность в точке $\mu = 0{,}5$ равна
\[
  \beta_\varphi(0{,}5) = \Prob_{0{,}5}(\bar X > 0{,}329) = \Prob\bigl(Z > (0{,}329 - 0{,}5)\sqrt{25}\bigr) = \Prob(Z > -0{,}855) \approx 0{,}804,
\]
где $Z \sim \mathcal{N}(0, 1)$. Значит, при $n = 25$ истинное значение среднего $0{,}5$ обнаруживается с вероятностью около~$0{,}80$.
\end{example}

\begin{example}[Подбор объёма выборки]
\label{ex:sample-size}
В тех же условиях найдём~$n$, при котором мощность в точке $\mu = 0{,}5$ равна~$0{,}95$. Условие
\[
  1 - \Phi\bigl(z_\alpha - \mu_1 \sqrt n / \sigma\bigr) = 0{,}95
\]
переписывается как $z_\alpha - 0{,}5 \sqrt n = -z_{0{,}05} = -1{,}645$, то есть $\sqrt n = (z_\alpha + 1{,}645)/0{,}5 = (1{,}645 + 1{,}645)/0{,}5 = 6{,}58$, откуда $n \geq 44$. Общая формула $n = ((z_\alpha + z_\beta)\sigma/(\mu_1 - \mu_0))^2$ суммирует это рассуждение.
\end{example}

\begin{problem}
\label{prob:bernoulli-test}
$X_1, \dots, X_n$~--- независимые одинаково распределённые $\mathrm{Bernoulli}(p)$. Покажите, что наиболее мощный критерий проверки $H_0 : p = 1/2$ против $H_1 : p = 3/4$ на уровне~$\alpha$ отвергает~$H_0$ при числе успехов $S = \sum X_i$, превышающем некоторый порог, и определите этот порог.
\end{problem}

\begin{proof}[Решение]
Функция правдоподобия равна $p^S (1 - p)^{n - S}$, поэтому отношение правдоподобия $L(\vect{x}) = (3/4)^S (1/4)^{n - S} / (1/2)^n = 2^n \cdot 3^S / 4^n = 3^S / 2^n$~--- строго возрастающая функция от~$S$. По лемме Неймана--Пирсона (\cref{thm:neyman-pearson}) НМ-критерий отвергает~$H_0$ при $S > c$ с целочисленным порогом~$c$, выбранным так, чтобы $\Prob_{p = 1/2}(S > c) = \alpha$, то есть~$c$~--- верхняя $\alpha$-критическая точка распределения $\mathrm{Bin}(n, 1/2)$. Например, при $n = 20$ и $\alpha = 0{,}05$ из биномиальной функции распределения находим $c = 14$ ($\Prob(S \geq 15 \mid p = 1/2) \approx 0{,}021$, тогда как $\Prob(S \geq 14 \mid p = 1/2) \approx 0{,}058$); рандомизация на множестве $\{S = 14\}$ позволяет точно настроить уровень $\alpha = 0{,}05$ и получить строго НМ-критерий. По тому же рассуждению о монотонном отношении правдоподобия этот критерий является РНМ для односторонней сложной альтернативы $p > 1/2$.
\end{proof}
