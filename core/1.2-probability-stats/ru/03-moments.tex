\section{Математическое ожидание, дисперсия, ковариация}
\label{sec:1.2.03-moments}

\begin{ticket}
Определение математического ожидания, дисперсии, ковариации и корреляции, их свойства.
\end{ticket}

\subsection*{Теория}

Всюду в этом разделе $(\Omega, \mathcal{F}, \Prob)$~--- фиксированное вероятностное пространство, а~$X, Y$~--- случайные величины на нём (\cref{def:random-variable}). Общее (лебеговское) построение математического ожидания на $L^1$ изложено в~\cite[гл.~II, \S~6]{Shiryaev1989}; ниже мы определяем дискретный и абсолютно непрерывный случаи и выписываем свойства, выполненные для обоих.

\begin{definition}[Математическое ожидание]
\label{def:expectation}
\emph{Математическим ожиданием} (\emph{средним значением}) случайной величины~$X$ называется одна из величин ниже; в каждом случае требуется, чтобы соответствующая сумма или интеграл с заменой подынтегрального выражения на его модуль был конечным (\emph{абсолютная сходимость}).

\emph{Дискретный случай.} Если~$X$ принимает счётное число значений $\{x_k\}_{k \geq 1}$ с вероятностями $p_k = \Prob(X = x_k)$, то
\[
  \E X \;=\; \sum_{k \geq 1} x_k\, p_k.
\]

\emph{Абсолютно непрерывный случай.} Если~$X$ имеет плотность~$f_X$, то есть $F_X(x) = \int_{-\infty}^{x} f_X(t)\, dt$, то
\[
  \E X \;=\; \int_{\R} x\, f_X(x)\, dx.
\]
Случайная величина~$X$ называется \emph{интегрируемой}, если $\E |X| < \infty$; пишут $X \in L^1$ при $\E |X| < \infty$ и $X \in L^2$ при $\E X^2 < \infty$.
\end{definition}

\begin{proposition}[Свойства математического ожидания]
\label{prop:exp-properties}
Для интегрируемых случайных величин $X, Y$ и констант $a, b \in \R$ верно:
\begin{enumerate}[label=(\alph*)]
  \item \emph{линейность:} $\E(aX + bY) = a\, \E X + b\, \E Y$;
  \item \emph{монотонность:} если $X \leq Y$ почти наверное, то $\E X \leq \E Y$;
  \item \emph{постоянная:} $\E(c) = c$ для любой константы~$c$;
  \item \emph{неравенство треугольника:} $|\E X| \leq \E |X|$;
  \item \emph{индикаторное правило:} для любого события~$A$ выполнено $\E\mathbf{1}_A = \Prob(A)$.
\end{enumerate}
\end{proposition}
\proofof{exp-properties}

\begin{proposition}[Закон бессознательного статистика]
\label{prop:lotus}
Пусть $g \colon \R \to \R$~--- борелевская функция.
\begin{itemize}
  \item Дискретный случай: $\E\, g(X) = \sum_k g(x_k)\, p_k$, если ряд сходится абсолютно.
  \item Абсолютно непрерывный случай: $\E\, g(X) = \int_{\R} g(x)\, f_X(x)\, dx$, если интеграл сходится абсолютно.
\end{itemize}
\end{proposition}
\proofof{lotus}

\medskip
\noindent\textbf{Дисперсия.} Дисперсия измеряет среднеквадратичное отклонение от среднего значения и служит канонической скалярной характеристикой разброса.

\begin{definition}[Дисперсия и среднеквадратичное отклонение]
\label{def:variance}
Для $X \in L^2$ \emph{дисперсия} случайной величины~$X$ задаётся равенством
\[
  \Var X \;=\; \E\bigl[(X - \E X)^2\bigr],
\]
а \emph{среднеквадратичное отклонение} равно $\sigma_X = \sqrt{\Var X}$.
\end{definition}

\begin{proposition}[Свойства дисперсии]
\label{prop:var-properties}
Для $X \in L^2$ и $a, b \in \R$:
\begin{enumerate}[label=(\alph*)]
  \item \emph{вычислительная формула:} $\Var X = \E X^2 - (\E X)^2$;
  \item \emph{неотрицательность:} $\Var X \geq 0$, причём $\Var X = 0$ тогда и только тогда, когда~$X$ почти наверное константа;
  \item \emph{аффинное масштабирование:} $\Var(aX + b) = a^2\, \Var X$;
  \item \emph{инвариантность относительно сдвига:} $\Var(X + b) = \Var X$.
\end{enumerate}
\end{proposition}
\proofof{var-properties}

\medskip
\noindent\textbf{Ковариация и коэффициент корреляции.} Ковариация распространяет дисперсию на пару $(X, Y)$ и улавливает линейную зависимость величин.

\begin{definition}[Ковариация]
\label{def:covariance}
Для $X, Y \in L^2$ \emph{ковариацией} случайных величин~$X$ и~$Y$ называется
\[
  \Cov(X, Y) \;=\; \E\bigl[(X - \E X)(Y - \E Y)\bigr] \;=\; \E(XY) - (\E X)(\E Y).
\]
Случайные величины называются \emph{некоррелированными}, если $\Cov(X, Y) = 0$.
\end{definition}

\begin{proposition}[Свойства ковариации]
\label{prop:cov-properties}
Для $X, Y, Z \in L^2$ и $a, b, c \in \R$:
\begin{enumerate}[label=(\alph*)]
  \item $\Cov(X, X) = \Var X$;
  \item \emph{симметрия:} $\Cov(X, Y) = \Cov(Y, X)$;
  \item \emph{билинейность:} $\Cov(aX + bY, Z) = a\, \Cov(X, Z) + b\, \Cov(Y, Z)$;
  \item \emph{константа выпадает:} $\Cov(X, c) = 0$;
  \item \emph{дисперсия суммы:} для $X_1, \dots, X_n \in L^2$
        \[
          \Var\!\Bigl(\sum_{i=1}^n X_i\Bigr) = \sum_{i=1}^n \Var X_i + 2 \!\!\sum_{1 \leq i < j \leq n}\!\! \Cov(X_i, X_j).
        \]
\end{enumerate}
\end{proposition}
\proofof{cov-properties}

\begin{theorem}[Коши--Буняковский]
\label{thm:cauchy-schwarz}
Для произвольных $X, Y \in L^2$
\[
  |\E(XY)| \;\leq\; \sqrt{\E X^2 \cdot \E Y^2},
\]
причём равенство достигается тогда и только тогда, когда $Y = \alpha X$ почти наверное при некотором $\alpha \in \R$ (либо одна из величин~$X, Y$ почти наверное равна нулю). Заменяя~$X, Y$ на $X - \E X$, $Y - \E Y$, получаем
\[
  |\Cov(X, Y)| \;\leq\; \sqrt{\Var X \cdot \Var Y}.
\]
\end{theorem}
\proofof{cauchy-schwarz}

\begin{definition}[Коэффициент корреляции]
\label{def:correlation}
Для $X, Y \in L^2$ с $\Var X > 0$ и $\Var Y > 0$ \emph{(пирсоновским) коэффициентом корреляции} называется
\[
  \rho(X, Y) \;=\; \frac{\Cov(X, Y)}{\sqrt{\Var X \cdot \Var Y}}.
\]
\end{definition}

\begin{corollary}[Свойства~$\rho$]
\label{cor:rho-properties}
$|\rho(X, Y)| \leq 1$, причём $\rho(X, Y) = \pm 1$ тогда и только тогда, когда $Y = aX + b$ почти наверное при некоторых $a, b \in \R$ (со знаком $\sgn a = \sgn \rho$). Корреляция инвариантна относительно положительного аффинного преобразования: $\rho(\alpha X + \beta, \gamma Y + \delta) = \rho(X, Y)$ при $\alpha, \gamma > 0$ (и переходит в $-\rho(X, Y)$ при $\alpha \gamma < 0$).
\end{corollary}
\proofof{rho-properties}

\begin{remark}
Коэффициент~$\rho$ улавливает только \emph{линейную} часть зависимости. Две величины могут быть некоррелированными, но при этом сильно зависимыми: например, для $X \sim \mathrm{Uniform}(-1, 1)$ и $Y = X^2$ имеем $\Cov(X, Y) = \E X^3 - \E X \cdot \E X^2 = 0$, но~$Y$ детерминированно зависит от~$X$. Роль корреляции в задаче линейного прогнозирования рассматривается в~\cref{sec:1.2.11-least-squares-gauss-markov}.
\end{remark}

\subsection*{Доказательства}

\begin{proof}[\Cref{prop:exp-properties}]
\proofanchor{exp-properties}
Линейность очевидна в обоих представлениях: в дискретном случае~--- через абсолютную сходимость и перестановку двойных рядов, в непрерывном~--- через линейность интеграла Лебега.

\emph{(b)} Если $X \leq Y$ почти наверное, то $Y - X \geq 0$ почти наверное; в дискретном случае сумма $\sum (y_k - x_k) p_k$ неотрицательна почленно, в непрерывном случае подынтегральная функция $(Y - X)(\omega) \geq 0$ почти наверное даёт неотрицательный интеграл. Из линейности $\E Y - \E X = \E(Y - X) \geq 0$.

\emph{(c)} Постоянная случайная величина $X \equiv c$ принимает единственное значение~$c$ с вероятностью~$1$, поэтому $\E X = c \cdot 1 = c$.

\emph{(d)} $X \leq |X|$ и $-X \leq |X|$ почти наверное, поэтому по~(b) $\pm \E X \leq \E |X|$, откуда $|\E X| \leq \E |X|$.

\emph{(e)} Индикатор $\mathbf{1}_A$ принимает значение~$1$ на~$A$ (с вероятностью $\Prob(A)$) и~$0$ на $\Omega \setminus A$, поэтому по дискретной формуле $\E \mathbf{1}_A = 1 \cdot \Prob(A) + 0 \cdot \Prob(\Omega \setminus A) = \Prob(A)$.
\end{proof}

\begin{proof}[\Cref{prop:lotus}]
\proofanchor{lotus}
Дискретный случай очевиден: $g(X)$ принимает значение $g(x_k)$ на событии $\{X = x_k\}$, поэтому по дискретной формуле из~\cref{def:expectation} имеем $\E g(X) = \sum_k g(x_k) \Prob(X = x_k)$.

В абсолютно непрерывном случае равенство сначала проверяется для индикаторных функций $g = \mathbf{1}_B$ (где оно сводится к $\Prob(X \in B) = \int_B f_X$~--- определяющему свойству плотности), затем по линейности продолжается на простые функции, а с простых на произвольные неотрицательные борелевские~$g$~--- по теореме Лебега о монотонной сходимости. Разложение $g = g^+ - g^-$ переносит результат на функции произвольного знака. Условие абсолютной сходимости гарантирует конечность обеих частей. Подробное изложение~--- в~\cite[гл.~II, \S~6]{Shiryaev1989}.
\end{proof}

\begin{proof}[\Cref{prop:var-properties}]
\proofanchor{var-properties}
\emph{(a)} Раскроем квадрат и применим линейность: $\E[(X - \E X)^2] = \E X^2 - 2\, \E X \cdot \E X + (\E X)^2 = \E X^2 - (\E X)^2$.

\emph{(b)} $(X - \E X)^2 \geq 0$ почти наверное, поэтому по монотонности математического ожидания его среднее неотрицательно. Равенство $\Var X = 0$ влечёт $(X - \E X)^2 = 0$ почти наверное, то есть $X = \E X$ почти наверное.

\emph{(c, d)} Положим $Y = aX + b$ и применим линейность: $\E Y = a\, \E X + b$ и $Y - \E Y = a(X - \E X)$, откуда $\Var Y = \E[a^2 (X - \E X)^2] = a^2\, \Var X$. Подстановка $a = 1$ даёт инвариантность относительно сдвига.
\end{proof}

\begin{proof}[\Cref{prop:cov-properties}]
\proofanchor{cov-properties}
\emph{(a, b)} Непосредственно из~\cref{def:covariance}.

\emph{(c)} Применим линейность математического ожидания к разложению $(aX + bY - (a \E X + b \E Y))(Z - \E Z)$.

\emph{(d)} $\Cov(X, c) = \E[(X - \E X)(c - c)] = 0$.

\emph{(e)} Применяя~(c) последовательно по обоим аргументам, получаем
\[
  \Var\!\Bigl(\sum_i X_i\Bigr) = \Cov\!\Bigl(\sum_i X_i,\, \sum_j X_j\Bigr) = \sum_{i, j} \Cov(X_i, X_j).
\]
Двойную сумму разобьём на диагональ ($i = j$, дающую слагаемое~$\Var X_i$) и недиагональные слагаемые; пары $(i, j)$ и $(j, i)$ объединим по симметрии в одну сумму по $i < j$ с коэффициентом~$2$.
\end{proof}

\begin{proof}[\Cref{thm:cauchy-schwarz}]
\proofanchor{cauchy-schwarz}
Если $\E X^2 = 0$, то из $X^2 \geq 0$ почти наверное и $\E X^2 = 0$ по монотонности математического ожидания (\cref{prop:exp-properties}\,(b), применённое к~$X^2$ и константе~$0$) следует $X^2 = 0$ почти наверное, а значит, и $X = 0$ почти наверное; тогда $XY = 0$ почти наверное, и обе части неравенства равны нулю. Аналогично при $\E Y^2 = 0$. В противном случае рассмотрим при $t \in \R$
\[
  0 \leq \E\bigl[(tX - Y)^2\bigr] = t^2\, \E X^2 - 2t\, \E(XY) + \E Y^2.
\]
Это неотрицательный квадратичный относительно~$t$ многочлен, поэтому его дискриминант неположителен: $4 (\E XY)^2 - 4\, \E X^2 \cdot \E Y^2 \leq 0$, откуда $(\E XY)^2 \leq \E X^2 \cdot \E Y^2$. Равенство в дискриминанте отвечает существованию вещественного корня $t = t_0$ многочлена $\E[(tX - Y)^2]$, что влечёт $Y = t_0 X$ почти наверное. Ковариационное неравенство получается применением только что доказанного к центрированным величинам $X - \E X$, $Y - \E Y$.
\end{proof}

\begin{proof}[\Cref{cor:rho-properties}]
\proofanchor{rho-properties}
Неравенство $|\rho(X, Y)| \leq 1$~--- это ковариационное неравенство Коши--Буняковского (\cref{thm:cauchy-schwarz}), делённое на $\sqrt{\Var X \cdot \Var Y}$. Равенство $|\rho| = 1$ соответствует $Y - \E Y = \alpha (X - \E X)$ почти наверное, то есть $Y = \alpha X + (\E Y - \alpha \E X)$ почти наверное. Аффинная инвариантность следует из равенств $\Cov(\alpha X + \beta, \gamma Y + \delta) = \alpha \gamma \Cov(X, Y)$ (\cref{prop:cov-properties}\,(c, d)) и $\Var(\alpha X + \beta) = \alpha^2 \Var X$ (\cref{prop:var-properties}\,(c)): множители $\alpha^2$ и $\gamma^2$ под корнем сокращают $\alpha \gamma$ в числителе, когда $\alpha \gamma > 0$.
\end{proof}

\subsection*{Примеры}

\begin{example}[Правильная игральная кость]
\label{ex:die-moments}
Пусть~$X \in \{1, \dots, 6\}$ распределена равномерно. Тогда $\E X = \tfrac{1}{6}(1 + 2 + 3 + 4 + 5 + 6) = 7/2$ и $\E X^2 = \tfrac{1}{6}(1 + 4 + 9 + 16 + 25 + 36) = 91/6$, поэтому $\Var X = 91/6 - 49/4 = 35/12$ и $\sigma_X = \sqrt{35/12} \approx 1{,}708$.
\end{example}

\begin{example}[Число выпадений ``орла'' в~$n$ бросках]
\label{ex:binomial-moments}
Пусть монета подбрасывается~$n$ раз независимо, вероятность выпадения ``орла'' равна~$p$; обозначим $X_i = \mathbf{1}_{\{\text{бросок } i \text{ выпал ``орлом''}\}}$ и $S_n = X_1 + \cdots + X_n$. Каждое~$X_i$~--- бернуллиевская величина с $\E X_i = p$ и $\Var X_i = p(1 - p)$. По линейности (независимость не используется!) $\E S_n = np$. Броски независимы, а потому попарно некоррелированы (это последнее замечание опирается на~\cref{thm:indep-multiplicative} из~\cref{sec:1.2.04-independence}); по~\cref{prop:cov-properties}\,(e) перекрёстные ковариации обнуляются, и $\Var S_n = n p (1 - p)$.
\end{example}

\begin{example}[Вычисление ковариации и коэффициента корреляции]
\label{ex:cov-pair}
Пусть пара $(X, Y)$ распределена равномерно на четырёх точках $\{(0, 0), (1, 0), (1, 1), (2, 1)\}$. Прямой подсчёт даёт $\E X = 1$, $\E Y = 1/2$, $\E XY = \tfrac{1}{4}(0 + 0 + 1 + 2) = 3/4$, поэтому $\Cov(X, Y) = 3/4 - 1 \cdot 1/2 = 1/4$. Также $\Var X = \tfrac{1}{4}(1 + 0 + 0 + 1) = 1/2$ и $\Var Y = \tfrac{1}{4}\bigl(\tfrac{1}{4} \cdot 4\bigr) = 1/4$, откуда $\rho(X, Y) = (1/4) / \sqrt{1/2 \cdot 1/4} = 1/\sqrt{2} \approx 0{,}707$.
\end{example}

\begin{example}[Некоррелированные, но зависимые: $Y = X^2$]
\label{ex:uncorrelated-dependent}
Пусть~$X$ распределена равномерно на $\{-1, 0, 1\}$ и $Y = X^2$. Тогда $\E X = 0$ и $\E XY = \E X^3 = 0$ (нечётная функция на симметричном распределении), поэтому $\Cov(X, Y) = 0$~--- хотя~$Y$ детерминированно зависит от~$X$. Это показывает, что некоррелированность строго слабее независимости: коэффициент корреляции улавливает только линейную составляющую зависимости.
\end{example}

\begin{problem}
\label{prob:variance-of-sum-correlated}
Пусть $X_1, X_2, X_3 \in L^2$~--- три случайные величины, каждая с дисперсией~$1$, попарные коэффициенты корреляции которых равны общему значению $\rho \in (-1, 1)$. Найдите $\Var(X_1 + X_2 + X_3)$ и значение~$\rho$, минимизирующее дисперсию суммы. Каково наименьшее допустимое значение~$\rho$?
\end{problem}

\begin{proof}[Решение]
По~\cref{prop:cov-properties}\,(e) $\Var(X_1 + X_2 + X_3) = 3 \cdot 1 + 2 \cdot 3 \cdot \rho = 3 + 6\rho$. Эта величина убывает по~$\rho$ и минимизируется при наименьшем допустимом~$\rho$. Но дисперсия неотрицательна, поэтому $3 + 6\rho \geq 0$, то есть $\rho \geq -1/2$. Граница достигается, например, на $X_1 = U_1 - U_2$, $X_2 = U_2 - U_3$, $X_3 = U_3 - U_1$ при некоррелированных $U_1, U_2, U_3$ с равными дисперсиями: их попарные ковариации равны~$-1$ в ненормированном виде и нормируются до $\rho = -1/2$. Общий факт: для $n$-набора наименьшее допустимое общее значение коэффициента корреляции равно $-1/(n - 1)$~--- при меньших значениях $n \times n$-матрица корреляций перестала бы быть неотрицательно определённой.
\end{proof}
