\section{Линейная зависимость и ранг}
\label{sec:1.1.02-rank}

\begin{ticket}
Линейная зависимость и ранг. Линейная зависимость строк (столбцов). Основная лемма о линейной зависимости, базис и ранг системы строк (столбцов). Ранг матрицы. Критерий совместности и определенности системы линейных уравнений в терминах рангов матриц. Фундаментальная система решений однородной системы линейных уравнений.
\end{ticket}

\subsection*{Теория}

Все рассмотрения по-прежнему ведутся над фиксированным полем~$\F$. Под
\emph{системой векторов} понимается конечный упорядоченный набор
$\vect{v}_1, \dots, \vect{v}_k \in \F^n$; интерес представляют системы
строк или столбцов матрицы.

\begin{definition}
\label{def:lin-dep}
\emph{Линейной комбинацией} векторов $\vect{v}_1, \dots, \vect{v}_k$ с
коэффициентами $\alpha_1, \dots, \alpha_k \in \F$ называется вектор
$\alpha_1 \vect{v}_1 + \dots + \alpha_k \vect{v}_k$. Комбинация
называется \emph{тривиальной}, если все $\alpha_i = 0$. Система
$\vect{v}_1, \dots, \vect{v}_k$ \emph{линейно зависима}, если некоторая
её нетривиальная линейная комбинация равна $\vect{0}$, и \emph{линейно
независима} в противном случае. Говорят, что вектор~$\vect{u}$
\emph{линейно выражается через} $\vect{v}_1, \dots, \vect{v}_k$, если
$\vect{u}$ есть какая-либо их линейная комбинация.
\end{definition}

\begin{remark}
\label{rem:basic-dep-facts}
Сразу отметим два простых факта: (i)~любая система, содержащая
вектор~$\vect{0}$, линейно зависима; (ii)~добавление новых векторов к
линейно зависимой системе оставляет её линейно зависимой, а удаление
векторов из линейно независимой системы оставляет её линейно
независимой.
\end{remark}

\begin{theorem}[Основная лемма о линейной зависимости]
\label{thm:main-lemma}
Если каждый из векторов $\vect{v}_1, \dots, \vect{v}_k$ линейно
выражается через $\vect{u}_1, \dots, \vect{u}_m$ и $k > m$, то система
$\vect{v}_1, \dots, \vect{v}_k$ линейно зависима.
\end{theorem}
\proofof{main-lemma}

\begin{remark}
\label{rem:main-lemma-abstract}
Доказательство использует лишь сложение векторов и умножение на скаляр,
поэтому утверждение и тот же ход рассуждений переносятся без изменений
на произвольное векторное пространство над~$\F$ (см.~\cref{sec:1.1.04-vector-spaces}).
\end{remark}

\begin{definition}
\label{def:basis-rank-system}
\emph{Базисом} системы $\vect{v}_1, \dots, \vect{v}_k$ называется её
линейно независимая подсистема $\vect{v}_{i_1}, \dots, \vect{v}_{i_r}$,
через которую линейно выражается каждый из $\vect{v}_j$. По основной
лемме все базисы данной системы содержат одно и то же число векторов;
это общее число называется \emph{рангом} системы и обозначается
$\rank(\vect{v}_1, \dots, \vect{v}_k)$.
\end{definition}

Для матрицы $\mat{A} \in \F^{m \times n}$ ранг системы её строк
называется \emph{ранг строк}, а ранг системы её столбцов~--- \emph{ранг
столбцов}.

\begin{theorem}[О совпадении ранга строк и ранга столбцов]
\label{thm:rank-row-col}
Для любой матрицы $\mat{A} \in \F^{m \times n}$ ранг строк и ранг
столбцов совпадают. Их общее значение называется \emph{рангом
матрицы}~$\mat{A}$ и обозначается $\rank \mat{A}$.
\end{theorem}
\proofof{rank-row-col}

\begin{proposition}[Сохранение ранга при ЭПС]
\label{prop:rank-ero-invariance}
Если матрица~$\mat{A}'$ получена из~$\mat{A}$ конечной
последовательностью элементарных преобразований строк, то
$\rank \mat{A}' = \rank \mat{A}$. В частности, $\rank \mat{A}$ равен
числу ведущих элементов в любом ступенчатом виде матрицы~$\mat{A}$.
\end{proposition}
\proofof{rank-ero-invariance}

\begin{theorem}[Кронекер--Капелли]
\label{thm:kronecker-capelli}
Система $\mat{A}\vect{x} = \vect{b}$ совместна тогда и только тогда,
когда $\rank \mat{A} = \rank [\mat{A} \mid \vect{b}]$. В случае
совместности решение единственно тогда и только тогда, когда
$\rank \mat{A} = n$ (где $n$~--- число неизвестных); иначе множество
решений зависит от $n - \rank \mat{A}$ свободных параметров.
\end{theorem}
\proofof{kronecker-capelli}

\begin{definition}
\label{def:fsr}
Система $\mat{A}\vect{x} = \vect{0}$ называется \emph{однородной}.
Множество её решений
\[
  \ker \mat{A} \;=\; \{\vect{x} \in \F^n : \mat{A}\vect{x} = \vect{0}\}
\]
является линейным подпространством в~$\F^n$ (замкнуто относительно
сложения и умножения на скаляр). \emph{Фундаментальной системой решений}
(ФСР) однородной системы $\mat{A}\vect{x} = \vect{0}$ называется любой
базис подпространства~$\ker \mat{A}$ в смысле
\cref{def:basis-rank-system}.
\end{definition}

\begin{theorem}[Структура множества решений]
\label{thm:solution-structure}
Пусть $r = \rank \mat{A}$.
\begin{enumerate}[label=(\alph*)]
  \item Подпространство $\ker \mat{A}$ имеет размерность $n - r$, и
        любая ФСР состоит из $n - r$ векторов.
  \item Если система $\mat{A}\vect{x} = \vect{b}$ совместна, а
        $\vect{x}_*$~--- какое-либо её решение, то всё множество
        решений имеет вид
        \[
          \{\vect{x}_* + c_1 \vect{f}_1 + \dots + c_{n-r} \vect{f}_{n-r}
          \;:\; c_1, \dots, c_{n-r} \in \F\},
        \]
        где $\vect{f}_1, \dots, \vect{f}_{n-r}$~--- произвольная ФСР
        ассоциированной однородной системы $\mat{A}\vect{x} = \vect{0}$.
\end{enumerate}
\end{theorem}
\proofof{solution-structure}

\begin{corollary}[Однозначность приведённого ступенчатого вида]
\label{cor:rref-unique}
Приведённый ступенчатый вид матрицы определён однозначно; это и есть
утверждение об однозначности из \cref{thm:gauss-rref}.
\end{corollary}
\proofof{rref-unique}

\begin{remark}
Альтернативные доказательства основной леммы и подробное обсуждение
понятия ранга см.\ в~\cite{Beklemishev2025}.
\end{remark}

\subsection*{Доказательства}

\begin{proof}[\Cref{thm:main-lemma}]
\proofanchor{main-lemma}
Индукция по~$m$.

\emph{База $m = 0$.} Пустая линейная комбинация равна $\vect{0}$,
поэтому каждый $\vect{v}_i = \vect{0}$. Тогда $1 \cdot \vect{v}_1 = \vect{0}$
есть нетривиальная комбинация, обращающаяся в нуль, и система
линейно зависима (она непуста, поскольку $k > m = 0$).

\emph{Шаг $m - 1 \Rightarrow m$.} Запишем
$\vect{v}_i = \sum_{j=1}^{m} c_{ij} \vect{u}_j$ для $i = 1, \dots, k$.

Если $c_{i1} = 0$ при всех~$i$, то каждый $\vect{v}_i$ принадлежит
линейной оболочке системы $\vect{u}_2, \dots, \vect{u}_m$ (всего
$m - 1$ векторов). Так как $k > m > m - 1$, по предположению индукции
векторы $\vect{v}_i$ линейно зависимы.

В противном случае, перенумеровав, считаем $c_{11} \neq 0$. Для
$i = 2, \dots, k$ положим
\[
  \vect{v}'_i \;=\; \vect{v}_i - \frac{c_{i1}}{c_{11}}\,\vect{v}_1.
\]
Коэффициент при $\vect{u}_1$ в $\vect{v}'_i$ обращается в нуль, поэтому
каждый $\vect{v}'_i$ есть линейная комбинация
$\vect{u}_2, \dots, \vect{u}_m$. Получили $k - 1$ векторов,
выраженных через $m - 1$ векторов, причём $k - 1 > m - 1$, так что по
предположению индукции $\vect{v}'_i$ линейно зависимы: существуют
$\lambda_2, \dots, \lambda_k$, не все равные нулю, такие, что
$\sum_{i \ge 2} \lambda_i \vect{v}'_i = \vect{0}$. Подставляя
определение $\vect{v}'_i$:
\[
  \sum_{i \ge 2} \lambda_i \vect{v}_i
  \;-\; \Bigl(\sum_{i \ge 2} \lambda_i \frac{c_{i1}}{c_{11}}\Bigr) \vect{v}_1
  \;=\; \vect{0}.
\]
Хотя бы один из $\lambda_i$ ($i \ge 2$) ненулевой, поэтому это
нетривиальная линейная комбинация исходных векторов
$\vect{v}_1, \dots, \vect{v}_k$.
\end{proof}

\begin{proof}[\Cref{prop:rank-ero-invariance}]
\proofanchor{rank-ero-invariance}
Каждое элементарное преобразование строк реализуется умножением слева
на обратимую матрицу~$\mat{E}$ (\cref{def:gauss-ero}). Строки
матрицы~$\mat{E}\mat{A}$ суть линейные комбинации строк~$\mat{A}$;
обратно, строки~$\mat{A} = \mat{E}^{-1}(\mat{E}\mat{A})$ суть линейные
комбинации строк~$\mat{E}\mat{A}$. Следовательно, линейные оболочки
строк (а значит, и ранги строк) у~$\mat{E}\mat{A}$ и~$\mat{A}$
совпадают.

Для столбцов: $j$-й столбец~$\mat{E}\mat{A}$ равен $\mat{E}\vect{c}_j$,
где $\vect{c}_j$~--- $j$-й столбец~$\mat{A}$. Поскольку~$\mat{E}$
обратима, векторы $\mat{E}\vect{c}_{j_1}, \dots, \mat{E}\vect{c}_{j_r}$
линейно независимы тогда и только тогда, когда таковы и
$\vect{c}_{j_1}, \dots, \vect{c}_{j_r}$. Поэтому ранг столбцов также
сохраняется.

Применим метод Гаусса и приведём~$\mat{A}$ к ступенчатому
виду~$\mat{U}$ с~$r$ ведущими элементами. Ненулевые строки
матрицы~$\mat{U}$ линейно независимы (расположение ведущих элементов
гарантирует, что любая нетривиальная их комбинация имеет ненулевой
элемент в столбце соответствующего ведущего) и порождают линейную
оболочку всех строк~$\mat{U}$. Следовательно, ранг строк~$\mat{U}$
равен~$r$, а потому и $\rank \mat{A} = r$.
\end{proof}

\begin{proof}[\Cref{thm:rank-row-col}]
\proofanchor{rank-row-col}
Приведём~$\mat{A}$ к ступенчатому виду~$\mat{U}$ с $r$ ведущими
элементами. По~\cref{prop:rank-ero-invariance} ранг строк~$\mat{A}$
равен~$r$, а ранг столбцов~$\mat{A}$ совпадает с рангом
столбцов~$\mat{U}$, так что достаточно показать, что ранг
столбцов~$\mat{U}$ равен~$r$.

Столбцы с ведущими элементами линейно независимы: в любой
нетривиальной их комбинации, рассматриваемой в строке самого нижнего
из участвующих ведущих, появляется ненулевая координата (остальные
участвующие ведущие столбцы расположены выше, и в выбранной строке
все они имеют нули в силу ступенчатой структуры: ниже ведущего элемента
в его столбце стоят только нули).
С другой стороны, каждый столбец~$\mat{U}$, не содержащий ведущего
элемента, выражается через ведущие столбцы, расположенные слева:
ненулевые координаты этого столбца лежат лишь в строках с ведущими
элементами в столбцах с меньшими номерами, так что обратная
подстановка из \cref{cor:gauss-solving} представляет его в их линейной
оболочке. Значит, ведущие столбцы образуют базис линейной оболочки
столбцов, и ранг столбцов~$\mat{U}$ равен~$r$.
\end{proof}

\begin{proof}[\Cref{thm:kronecker-capelli}]
\proofanchor{kronecker-capelli}
Система $\mat{A}\vect{x} = \vect{b}$ совместна тогда и только тогда,
когда $\vect{b}$ принадлежит линейной оболочке столбцов~$\mat{A}$.
Добавление столбца~$\vect{b}$ к матрице $[\mat{A} \mid \vect{b}]$ не
меняет линейную оболочку столбцов в точности тогда, когда $\vect{b}$
уже в неё входит; равносильно,
$\rank \mat{A} = \rank [\mat{A} \mid \vect{b}]$.

Единственность решения в случае совместности следует из
\cref{thm:solution-structure}\,(б): множество решений есть сдвиг
подпространства~$\ker \mat{A}$, поэтому состоит из единственной точки
тогда и только тогда, когда $\dim \ker \mat{A} = 0$, т.\,е.\ когда
$n - r = 0$, или $\rank \mat{A} = n$.
\end{proof}

\begin{proof}[\Cref{thm:solution-structure}]
\proofanchor{solution-structure}
\emph{(а)} Приведём~$\mat{A}$ к приведённому ступенчатому
виду~$\mat{U}$ (это сохраняет множество решений по
\cref{thm:gauss-invariance} и ранг по \cref{prop:rank-ero-invariance}).
Пусть $j_1 < \dots < j_r$~--- столбцы с ведущими элементами, а
$\ell_1 < \dots < \ell_{n-r}$~--- свободные столбцы. Для каждого
$s \in \{1, \dots, n-r\}$ определим $\vect{f}_s \in \F^n$ условиями
\[
  (\vect{f}_s)_{\ell_s} = 1, \quad
  (\vect{f}_s)_{\ell_t} = 0 \text{ при } t \neq s,
\]
а координаты $(\vect{f}_s)_{j_k}$ в базисных столбцах находятся
обратной подстановкой из уравнения $\mat{U}\vect{f}_s = \vect{0}$.

Векторы $\vect{f}_1, \dots, \vect{f}_{n-r}$ принадлежат~$\ker \mat{A}$
по построению. Они линейно независимы: ограничение соотношения
$\sum_s c_s \vect{f}_s = \vect{0}$ на свободную координату~$\ell_t$
даёт $c_t = 0$. Они порождают $\ker \mat{A}$: для произвольного
$\vect{x} \in \ker \mat{A}$ разность
$\vect{x} - \sum_s x_{\ell_s}\vect{f}_s$ имеет нули во всех свободных
координатах, а значит, по обратной подстановке нули и в базисных, то
есть совпадает с~$\vect{0}$. Таким образом, $\vect{f}_1, \dots,
\vect{f}_{n-r}$~--- базис подпространства~$\ker \mat{A}$ из $n - r$
векторов.

\emph{(б)} Если $\vect{x}_*$~--- частное решение, а $\vect{x}$~---
произвольное решение, то
$\mat{A}(\vect{x} - \vect{x}_*) = \vect{b} - \vect{b} = \vect{0}$,
откуда $\vect{x} - \vect{x}_* \in \ker \mat{A}$. Обратно,
$\vect{x}_* + \vect{y}$ удовлетворяет системе при любом
$\vect{y} \in \ker \mat{A}$. Полное множество решений равно
$\vect{x}_* + \ker \mat{A}$, а по части~(а) совпадает с
$\vect{x}_* + \spn(\vect{f}_1, \dots, \vect{f}_{n-r})$.
\end{proof}

\begin{proof}[\Cref{cor:rref-unique}]
\proofanchor{rref-unique}
Пусть $\mat{U}$ и $\mat{U}'$~--- две приведённые ступенчатые матрицы,
обе строчно эквивалентные~$\mat{A}$. По
\cref{prop:rank-ero-invariance} линейные оболочки их строк совпадают
и равны~$V$. Покажем, что $\mat{U} = \mat{U}'$.

Столбец с номером~$j$ является базисным столбцом приведённого
ступенчатого вида тогда и только тогда, когда в~$V$ найдётся элемент,
у которого первая ненулевая координата стоит ровно в столбце~$j$.
В одну сторону: ведущий элемент $k$-й строки приведённого ступенчатого
вида находится в позиции $j_k$, а левее него стоят нули, поэтому сама
строка~--- такой элемент. Обратно: пусть $v \in V$ имеет первую
ненулевую координату в столбце~$j$ и пусть $v = \sum_k c_k r_k$, где
$r_k$~--- $k$-я строка~$\mat U$. Тогда $c_k = v[j_k]$ (ибо $r_k[j_l] =
\delta_{kl}$), и из условия $v[i] = 0$ при $i < j$ последовательно
получаем $c_k = 0$ для всех~$k$ с $j_k < j$; значит, $v$~--- комбинация
строк с $j_k \ge j$, и её первая ненулевая координата лежит в столбце
$\min\{j_k : c_k \neq 0\} \ge j$, что вынуждает $j$ быть одним из~$j_k$.

Это условие зависит лишь от~$V$, поэтому базисные столбцы у~$\mat{U}$
и~$\mat{U}'$ одни и те же: $j_1 < \dots < j_r$.

В приведённом ступенчатом виде $k$-я ненулевая строка~--- единственный
элемент~$V$, имеющий $1$ в позиции~$j_k$, нули в позициях
$j_1, \dots, j_{k-1}, j_{k+1}, \dots, j_r$ и нули во всех позициях
$1, \dots, j_k - 1$. (Последнее условие заставляет носитель лежать в
столбцах $\ge j_k$, после чего нормировка приведённого вида
определяет все коэффициенты.) Единственность такого элемента в~$V$
влечёт $\mat{U} = \mat{U}'$.
\end{proof}

\subsection*{Примеры}

\begin{example}[Вычисление ранга методом Гаусса]
\label{ex:rank-compute}
Найдём ранг матрицы
\[
  \mat{A} = \begin{pmatrix}
    1 & 2 & 3 \\
    2 & 4 & 6 \\
    1 & 1 & 1
  \end{pmatrix}.
\]
Применим $R_2 \to R_2 - 2R_1$ и $R_3 \to R_3 - R_1$:
\[
  \mat{A} \;\longrightarrow\;
  \begin{pmatrix}
    1 & 2 & 3 \\
    0 & 0 & 0 \\
    0 & -1 & -2
  \end{pmatrix}
  \;\xrightarrow{R_2 \leftrightarrow R_3}\;
  \begin{pmatrix}
    1 & 2 & 3 \\
    0 & -1 & -2 \\
    0 & 0 & 0
  \end{pmatrix}.
\]
В ступенчатом виде два ведущих элемента, поэтому $\rank \mat{A} = 2$.
Вторая строка исходной матрицы вдвое больше первой; именно поэтому
ведущий элемент в этой строке отсутствует.
\end{example}

\begin{example}[Кронекер--Капелли и общее решение]
\label{ex:kc-general-solution}
Для системы
\[
  \begin{aligned}
    x + y - z &= 1, \\
    2x + y + z &= 3, \\
    x + 2y - 4z &= 0
  \end{aligned}
\]
расширенная матрица преобразованиями $R_2 \to R_2 - 2R_1$,
$R_3 \to R_3 - R_1$, $R_3 \to R_3 + R_2$ приводится к виду
\[
  \left[\begin{array}{rrr|r}
    1 & 1 & -1 & 1 \\
    0 & -1 & 3 & 1 \\
    0 & 0 & 0 & 0
  \end{array}\right].
\]
И блок коэффициентов, и расширенная матрица имеют по два ведущих
элемента, поэтому $\rank \mat{A} = \rank [\mat{A} \mid \vect{b}] = 2$,
и по \cref{thm:kronecker-capelli} система совместна. При $n = 3$
неизвестных и $r = 2$ имеем $\dim \ker \mat{A} = 1$, то есть множество
решений~--- прямая.

Возьмём $z$ за свободную переменную: из второй строки $y = 3z - 1$,
из первой $x = 2 - 2z$. Итак,
\[
  \begin{pmatrix} x \\ y \\ z \end{pmatrix}
  =
  \begin{pmatrix} 2 \\ -1 \\ 0 \end{pmatrix}
  + z \begin{pmatrix} -2 \\ 3 \\ 1 \end{pmatrix},
  \qquad z \in \F.
\]
Вектор $(2, -1, 0)\T$~--- частное решение, а $(-2, 3, 1)\T$~---
ФСР ассоциированной однородной системы (с точностью до скаляра).
\end{example}

\begin{example}[Фундаментальная система решений]
\label{ex:fsr}
Найдём ФСР однородной системы
\[
  \begin{aligned}
    x + y - z + w &= 0, \\
    2x + 3y - z + 2w &= 0, \\
    3x + 5y - z + 3w &= 0.
  \end{aligned}
\]
Матрица коэффициентов приводится к ступенчатому виду:
\[
  \begin{pmatrix}
    1 & 1 & -1 & 1 \\
    2 & 3 & -1 & 2 \\
    3 & 5 & -1 & 3
  \end{pmatrix}
  \;\longrightarrow\;
  \begin{pmatrix}
    1 & 1 & -1 & 1 \\
    0 & 1 & 1 & 0 \\
    0 & 0 & 0 & 0
  \end{pmatrix}.
\]
Ведущие элементы в столбцах~$1$ и~$2$; $z$ и $w$~--- свободные. Из
второй строки $y = -z$, далее $x = -y + z - w = 2z - w$. Полагая
последовательно $(z, w) = (1, 0)$ и $(z, w) = (0, 1)$, получаем ФСР
\[
  \vect{f}_1 = \begin{pmatrix} 2 \\ -1 \\ 1 \\ 0 \end{pmatrix}, \qquad
  \vect{f}_2 = \begin{pmatrix} -1 \\ 0 \\ 0 \\ 1 \end{pmatrix}.
\]
Общее решение однородной системы~--- $c_1 \vect{f}_1 + c_2 \vect{f}_2$
при $c_1, c_2 \in \F$. Если бы правая часть была $(1, 3, 5)$, то
обратной подстановкой найдётся частное решение
$\vect{x}_* = (0, 1, 0, 0)\T$, и общее решение неоднородной системы
запишется в виде $\vect{x}_* + c_1 \vect{f}_1 + c_2 \vect{f}_2$~---
иллюстрация \cref{thm:solution-structure}\,(б).
\end{example}

\begin{problem}
\label{prob:rank-parameter}
При каких значениях $k \in \R$ матрица
\[
  \mat{A}(k) = \begin{pmatrix}
    k & 1 & 0 \\
    0 & k & 1 \\
    1 & 0 & k
  \end{pmatrix}
\]
имеет ранг~$3$, а при каких~--- ранг~$2$?
\end{problem}

\begin{proof}[Решение]
Раскладывая определитель по первому столбцу,
$\det \mat{A}(k) = k(k^2 - 0) + 1\,(1 \cdot 1 - 0) = k^3 + 1$.
Следовательно, $\rank \mat{A}(k) = 3$ тогда и только тогда, когда
$k^3 + 1 \neq 0$, то есть $k \neq -1$.

Остаётся убедиться, что $\rank \mat{A}(-1) = 2$ (ниже~$2$ ранг упасть
не может, так как не все $2 \times 2$-миноры обращаются в нуль). При
$k = -1$ имеем
\[
  \mat{A}(-1) =
  \begin{pmatrix}
    -1 & 1 & 0 \\
    0 & -1 & 1 \\
    1 & 0 & -1
  \end{pmatrix}
  \;\xrightarrow{R_3 \to R_3 + R_1 + R_2}\;
  \begin{pmatrix}
    -1 & 1 & 0 \\
    0 & -1 & 1 \\
    0 & 0 & 0
  \end{pmatrix},
\]
откуда $\rank \mat{A}(-1) = 2$.
\end{proof}
