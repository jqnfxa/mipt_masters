\section{Билинейные и квадратичные формы. Критерий Сильвестра}
\label{sec:1.1.07-quadratic-forms}

\begin{ticket}
Билинейные формы и их матрицы. Преобразование матрицы билинейной формы при замене базиса. Квадратичные формы. Матрица квадратичной формы. Метод Лагранжа приведения квадратичной формы к каноническому виду. Условие положительной (отрицательной) определенности квадратичной формы. Критерий Сильвестра.
\end{ticket}

\subsection*{Теория}

Все рассмотрения по-прежнему ведутся над фиксированным полем~$\F$.
Начиная с \cref{def:char-poly,thm:rank-row-col} мы свободно
пользовались техникой транспонирования и произведения матриц; теперь
применим её к билинейным формам. При обращении к поляризационному
тождеству дополнительно предполагается, что
$\operatorname{char} \F \neq 2$ (чтобы было определено деление на~$2$);
теоремы о приведении к каноническому виду затем специализируются к
случаю $\F = \R$ для обсуждения определённости в конце раздела.

\begin{definition}
\label{def:bilinear-form}
\emph{Билинейной формой} на векторном пространстве~$V$ над~$\F$
называется отображение $B \colon V \times V \to \F$, линейное по
каждому аргументу:
\[
  B(\alpha \vect{u} + \beta \vect{u}', \vect{v}) = \alpha B(\vect{u}, \vect{v}) + \beta B(\vect{u}', \vect{v}),
  \quad
  B(\vect{u}, \alpha \vect{v} + \beta \vect{v}') = \alpha B(\vect{u}, \vect{v}) + \beta B(\vect{u}, \vect{v}').
\]
Форма~$B$ называется \emph{симметричной}, если $B(\vect{u}, \vect{v}) = B(\vect{v}, \vect{u})$
для всех $\vect{u}, \vect{v}$.
\end{definition}

\begin{definition}
\label{def:matrix-bilinear}
Пусть $\{\vect{e}_1, \dots, \vect{e}_n\}$~--- базис~$V$.
\emph{Матрицей формы~$B$ в этом базисе} называется
$\mat{A} = (a_{ij}) \in \F^{n \times n}$ с $a_{ij} = B(\vect{e}_i, \vect{e}_j)$.
Для $\vect{u} = \sum x_i \vect{e}_i$ и $\vect{v} = \sum y_j \vect{e}_j$
имеет место равенство
\[
  B(\vect{u}, \vect{v}) \;=\; \sum_{i, j} a_{ij}\, x_i\, y_j \;=\; \vect{x}\T \mat{A}\, \vect{y}.
\]
Форма~$B$ симметрична тогда и только тогда, когда её матрица~$\mat{A}$
симметрична ($\mat{A}\T = \mat{A}$).
\end{definition}

\begin{theorem}[Биекция формы $\leftrightarrow$ матрицы в фиксированном базисе]
\label{thm:bilinear-matrix-bijection}
Для всякого фиксированного базиса~$V$ отображение, ставящее в
соответствие форме~$B$ её матрицу, есть биекция между билинейными
формами на~$V$ и матрицами из~$\F^{n \times n}$. Симметричным формам
соответствуют симметричные матрицы.
\end{theorem}
\begin{proof}
\emph{Инъективность.} Билинейная форма однозначно определяется своими
значениями на парах базисных векторов: для произвольных
$\vect{u} = \sum x_i \vect{e}_i$ и $\vect{v} = \sum y_j \vect{e}_j$
билинейность даёт
$B(\vect{u}, \vect{v}) = \sum_{i, j} x_i y_j\, B(\vect{e}_i, \vect{e}_j)
= \sum_{i, j} x_i y_j\, a_{ij}$
(\cref{def:matrix-bilinear}). Значит, две формы с одинаковой матрицей
совпадают всюду.

\emph{Сюръективность.} Для любой $\mat{A} = (a_{ij}) \in \F^{n \times n}$
формула
$B(\vect{u}, \vect{v}) := \sum_{i, j} a_{ij}\, x_i y_j$
определяет билинейную форму на~$V$, и значения $B(\vect{e}_i, \vect{e}_j) = a_{ij}$
показывают, что её матрица совпадает с~$\mat{A}$.

Симметричность формы~$B$ означает $a_{ij} = B(\vect{e}_i, \vect{e}_j) = B(\vect{e}_j, \vect{e}_i) = a_{ji}$,
то есть симметричность матрицы~$\mat{A}$.
\end{proof}

\begin{theorem}[Замена базиса: конгруэнтность]
\label{thm:bilinear-change-of-basis}
Если $\mat{A}$~--- матрица формы~$B$ в базисе $\{\vect{e}_j\}$, а
$\mat{T}$~--- матрица перехода к новому базису $\{\vect{e}'_j\}$
(\cref{def:transition-matrix}), то матрица формы~$B$ в новом базисе
равна
\[
  \mat{A}' \;=\; \mat{T}\T \mat{A}\, \mat{T}.
\]
Две матрицы, связанные таким соотношением, называются
\emph{конгруэнтными}.
\end{theorem}
\proofof{bilinear-change-of-basis}

\begin{remark}
Сравните с \cref{thm:operator-change-of-basis} для оператора: там
$\mat{A}' = \mat{T}^{-1} \mat{A}\, \mat{T}$ (подобие). Операторы и
билинейные формы~--- объекты разной природы: операторы переводят
векторы в векторы, формы переводят пары векторов в скаляры,~--- поэтому
при замене базиса они преобразуются по-разному. В частности,
$\det \mat{A}$~--- инвариант оператора по подобию, но для билинейной
формы~--- инвариант класса конгруэнтности лишь с точностью до
квадратного множителя (поскольку
$\det \mat{A}' = (\det \mat{T})^2 \det \mat{A}$).
\end{remark}

\begin{definition}
\label{def:quadratic-form}
\emph{Квадратичной формой} на~$V$ называется отображение
$Q \colon V \to \F$, для которого существует симметричная билинейная
форма~$B$ с $Q(\vect{v}) = B(\vect{v}, \vect{v})$ при любом~$\vect{v}$.
В этом случае говорят, что~$Q$ \emph{ассоциирована} с~$B$.
\emph{Матрицей формы~$Q$} (в фиксированном базисе) называется
симметричная матрица ассоциированной с ней~$B$.
\end{definition}

\begin{theorem}[Поляризация]
\label{thm:polarization}
Пусть $\operatorname{char} \F \neq 2$. Отображение
$B \mapsto Q(\vect{v}) := B(\vect{v}, \vect{v})$ есть биекция между
симметричными билинейными формами на~$V$ и квадратичными формами
на~$V$. Обратное отображение задаётся \emph{поляризационным
тождеством}
\[
  B(\vect{u}, \vect{v}) \;=\; \tfrac{1}{2}\bigl(Q(\vect{u} + \vect{v}) - Q(\vect{u}) - Q(\vect{v})\bigr).
\]
\end{theorem}
\proofof{polarization}

Перейдём к случаю $\F = \R$ для обсуждения определённости.

\begin{definition}
\label{def:definite}
Квадратичная форма~$Q$ на вещественном векторном пространстве~$V$
называется
\begin{itemize}
  \item \emph{положительно определённой}, если $Q(\vect{v}) > 0$ при $\vect{v} \neq \vect{0}$;
  \item \emph{отрицательно определённой}, если $Q(\vect{v}) < 0$ при $\vect{v} \neq \vect{0}$;
  \item \emph{положительно полуопределённой}, если $Q(\vect{v}) \ge 0$ при всех $\vect{v}$;
  \item \emph{неопределённой}, если~$Q$ принимает как строго положительные, так и строго отрицательные значения.
\end{itemize}
\end{definition}

\begin{theorem}[Метод Лагранжа]
\label{thm:lagrange-method}
Пусть $Q$~--- квадратичная форма на~$V$ над~$\F$ и
$\operatorname{char} \F \neq 2$. Тогда в~$V$ существует базис
$\{\vect{e}'_j\}$, в котором~$Q$ имеет \emph{канонический вид}
\[
  Q(\vect{v}) \;=\; \sum_{i=1}^{n} \lambda_i\, y_i^{2},
\]
где $\vect{y}$~--- координатный столбец~$\vect{v}$ в~$\{\vect{e}'_j\}$,
а скаляры $\lambda_i \in \F$ определяются алгоритмом из доказательства.
Эквивалентно, в новом базисе матрица формы~$Q$ становится диагональной
$\diag(\lambda_1, \dots, \lambda_n)$.
\end{theorem}
\proofof{lagrange-method}

\begin{theorem}[Закон инерции Сильвестра]
\label{thm:inertia}
При $\F = \R$ числа $n_+$ положительных и $n_-$ отрицательных
коэффициентов в любом каноническом виде вещественной квадратичной
формы~$Q$ зависят только от~$Q$, но не от выбранного базиса. Пара
$(n_+, n_-)$ называется \emph{сигнатурой} формы~$Q$, причём
$n_+ + n_- = \rank Q$ (ранг матрицы формы~$Q$).
\end{theorem}
\proofof{inertia}

\begin{theorem}[Критерий Сильвестра]
\label{thm:sylvester}
Пусть $\mat{A} \in \R^{n \times n}$~--- матрица квадратичной
формы~$Q$ в некотором базисе, а \emph{угловыми (ведущими) минорами}
матрицы~$\mat{A}$ называются числа
\[
  M_k \;=\; \det \begin{pmatrix} a_{11} & \cdots & a_{1k} \\ \vdots & & \vdots \\ a_{k1} & \cdots & a_{kk} \end{pmatrix},
  \qquad k = 1, 2, \dots, n.
\]
Тогда
\begin{enumerate}[label=(\alph*)]
  \item $Q$ положительно определена тогда и только тогда, когда $M_k > 0$ при всех $k = 1, \dots, n$;
  \item $Q$ отрицательно определена тогда и только тогда, когда $(-1)^k M_k > 0$ при всех $k = 1, \dots, n$
        (то есть знаки $M_k$ чередуются, начиная с $M_1 < 0$).
\end{enumerate}
\end{theorem}
\proofof{sylvester}

\begin{remark}
Альтернативное доказательство закона инерции
(\cref{thm:inertia}) использует спектральную теорему
(\cref{thm:spectral}): матрица вещественной симметричной формы
диагонализуется в ортонормированном базисе с вещественными собственными
значениями $\mu_1, \dots, \mu_n$, и $(n_+, n_-)$~--- это число
положительных и число отрицательных $\mu_i$, оба инвариантны. Тесно
связанный метод Якоби (альтернатива методу Лагранжа, дающая
диагональные элементы как отношения угловых миноров) обсуждается
в~\cite{Beklemishev2025}.
\end{remark}

\subsection*{Доказательства}

\begin{proof}[\Cref{thm:bilinear-change-of-basis}]
\proofanchor{bilinear-change-of-basis}
По \cref{thm:change-of-basis} координатные столбцы преобразуются как
$\vect{x} = \mat{T} \vect{x}'$ и $\vect{y} = \mat{T} \vect{y}'$.
Подставляя в матричное выражение для~$B$,
\[
  B(\vect{u}, \vect{v}) \;=\; \vect{x}\T \mat{A}\, \vect{y}
  \;=\; (\mat{T}\vect{x}')\T \mat{A}\, (\mat{T}\vect{y}')
  \;=\; {\vect{x}'}\T (\mat{T}\T \mat{A}\, \mat{T})\, {\vect{y}'}.
\]
По \cref{thm:bilinear-matrix-bijection} матрица формы~$B$ в новом
базисе однозначно определяется этим выражением в координатах, поэтому
она равна $\mat{T}\T \mat{A}\, \mat{T}$.
\end{proof}

\begin{proof}[\Cref{thm:polarization}]
\proofanchor{polarization}
\emph{Сюръективность.} Для всякой симметричной~$B$ функция
$Q(\vect{v}) = B(\vect{v}, \vect{v})$ по определению является
квадратичной формой.

\emph{Инъективность (и формула).} Пусть $B$ симметрична и
$Q(\vect{v}) = B(\vect{v}, \vect{v})$. Раскроем $Q(\vect{u} + \vect{v})$:
\[
  Q(\vect{u} + \vect{v}) = B(\vect{u} + \vect{v}, \vect{u} + \vect{v})
  = B(\vect{u}, \vect{u}) + B(\vect{u}, \vect{v}) + B(\vect{v}, \vect{u}) + B(\vect{v}, \vect{v})
  = Q(\vect{u}) + 2\,B(\vect{u}, \vect{v}) + Q(\vect{v}),
\]
здесь использована симметрия~$B$. Решая относительно~$B$, получаем
$B(\vect{u}, \vect{v}) = \tfrac{1}{2}\bigl(Q(\vect{u} + \vect{v}) - Q(\vect{u}) - Q(\vect{v})\bigr)$
(деление на~$2$ опирается на $\operatorname{char} \F \neq 2$). Значит,
$B$ восстанавливается по~$Q$ единственным образом, то есть отображение
$B \mapsto Q$ инъективно. Обратно, формула справа задаёт
симметричную билинейную форму для любой квадратичной~$Q$ (рутинная
проверка), что и определяет обратное отображение.
\end{proof}

\begin{proof}[\Cref{thm:lagrange-method}]
\proofanchor{lagrange-method}
Индукция по $n = \dim V$. Случай $n = 0$ тривиален; случай $n = 1$
тоже тривиален ($Q = a_{11} y_1^2$ в единственном базисе, возможно с
$a_{11} = 0$).

\emph{Шаг.} Запишем $Q = \sum a_{ij} x_i x_j$ в текущем базисе, при
этом $a_{ij} = a_{ji}$. Различаем три случая.

\emph{Случай 1: некоторое $a_{ii} \neq 0$.} Перенумеровав базисные
векторы, можно считать, что $a_{11} \neq 0$. Выделим полный квадрат
по~$x_1$:
\[
  Q = a_{11}\,x_1^2 + 2 \sum_{j \ge 2} a_{1j}\, x_1 x_j + \sum_{i, j \ge 2} a_{ij} x_i x_j
   = a_{11}\Bigl(x_1 + \sum_{j \ge 2} \tfrac{a_{1j}}{a_{11}}\,x_j\Bigr)^2 + Q'(x_2, \dots, x_n),
\]
где $Q'$~--- квадратичная форма от $n - 1$ переменных
$x_2, \dots, x_n$. Замена
\[
  y_1 = x_1 + \sum_{j \ge 2} \tfrac{a_{1j}}{a_{11}}\, x_j, \qquad y_j = x_j \text{ при } j \ge 2,
\]
невырождена (верхнетреугольная с единицами на диагонали). Применив
предположение индукции к~$Q'$, найдём дальнейшую замену в
$y_2, \dots, y_n$, приводящую $Q'$ к диагональному виду
$\sum_{j \ge 2} \mu_j z_j^2$. В совокупности получаем канонический вид
$a_{11} y_1^2 + \sum_{j \ge 2} \mu_j z_j^2$.

\emph{Случай 2: все $a_{ii} = 0$, но некоторое $a_{ij} \neq 0$ при $i \neq j$.}
Перенумеровав, считаем $a_{12} \neq 0$. Замена
\[
  x_1 = y_1 + y_2, \qquad x_2 = y_1 - y_2, \qquad x_k = y_k \text{ при } k \ge 3
\]
невырождена (обратная существует благодаря $\operatorname{char} \F \neq 2$).
Внедиагональный член
\[
  2 a_{12} x_1 x_2 = 2 a_{12} (y_1 + y_2)(y_1 - y_2) = 2 a_{12} (y_1^2 - y_2^2)
\]
порождает диагональные коэффициенты $2 a_{12}$ и $-2 a_{12}$ при
$y_1$ и $y_2$, и мы возвращаемся в Случай~1.

\emph{Случай 3: все $a_{ij} = 0$.} Тогда $Q$ тождественно равна нулю
и уже имеет канонический вид (все $\lambda_i = 0$).

Рекурсия завершается не более чем за $n$ шагов и даёт требуемый
канонический вид.
\end{proof}

\begin{proof}[\Cref{thm:inertia}]
\proofanchor{inertia}
Пусть~$Q$ имеет канонические виды в двух базисах:
\[
  Q(\vect{v}) = \sum_{i=1}^{n_+} y_i^2 - \sum_{i=n_+ + 1}^{n_+ + n_-} y_i^2,
  \qquad
  Q(\vect{v}) = \sum_{i=1}^{n'_+} (y'_i)^2 - \sum_{i=n'_+ + 1}^{n'_+ + n'_-} (y'_i)^2,
\]
с координатами~$\vect{y}$ в базисе $\{\vect{e}_i\}$ и~$\vect{y}'$ в
базисе $\{\vect{e}'_i\}$ (нулевые слагаемые здесь опущены, общее число
координат~$n$). Покажем, что $n_+ = n'_+$; симметричное рассуждение
даёт $n_- = n'_-$.

\emph{Подпространство положительности.} Положим
$P = \spn(\vect{e}_1, \dots, \vect{e}_{n_+})$, так что $\dim P = n_+$.
Для $\vect{v} = \sum_{i \le n_+} y_i \vect{e}_i$
выполнено $Q(\vect{v}) = \sum_{i \le n_+} y_i^2 \ge 0$, причём равенство
возможно лишь при $\vect{v} = \vect{0}$. На~$P$ форма~$Q$ положительно
определена.

\emph{Подпространство неположительности (в новом базисе).} Положим
$N' = \spn(\vect{e}'_{n'_+ + 1}, \dots, \vect{e}'_n)$, так что
$\dim N' = n - n'_+$. Для $\vect{v} = \sum_{i > n'_+} y'_i \vect{e}'_i$
имеем $Q(\vect{v}) = -\sum_{n'_+ < i \le n'_+ + n'_-} (y'_i)^2 \le 0$.

\emph{Пересечение.} Если $\vect{v} \in P \cap N'$, то~$Q(\vect{v})$
одновременно неотрицателен (как элемент~$P$) и неположителен (как
элемент~$N'$), значит, $Q(\vect{v}) = 0$. Положительная определённость
на~$P$ влечёт $\vect{v} = \vect{0}$. Поэтому $P \cap N' = \{\vect{0}\}$.

\emph{Неравенство размерностей.} По \cref{thm:grassmann}
$\dim P + \dim N' = \dim(P + N') + \dim(P \cap N') = \dim(P + N') \le n$,
откуда $n_+ + (n - n'_+) \le n$, то есть $n_+ \le n'_+$. Меняя ролями
два базиса, получаем $n'_+ \le n_+$, и тем самым $n_+ = n'_+$.

\emph{Инвариантность ранга.} По
\cref{thm:bilinear-change-of-basis} матрица формы~$Q$ в любом базисе
имеет вид $\mat{T}\T \mat{A}\, \mat{T}$ при обратимой~$\mat{T}$, и
$\rank(\mat{T}\T \mat{A}\, \mat{T}) = \rank \mat{A}$. В каноническом
виде ранг матрицы~$\diag(\lambda_1, \dots, \lambda_n)$ равен числу
ненулевых~$\lambda_i$, то есть $n_+ + n_-$. Значит, $\rank Q = n_+ + n_-$
зависит только от~$Q$, и установленное равенство $n_+ = n'_+$ влечёт
$n_- = n'_-$.
\end{proof}

\begin{proof}[\Cref{thm:sylvester}]
\proofanchor{sylvester}
Достаточно доказать пункт~(а); пункт~(б) следует из него применением к
$-Q$ с матрицей~$-\mat{A}$, у которой угловые миноры равны $(-1)^k M_k$.

\emph{(а)~$\Rightarrow$.} Пусть~$Q$ положительно определена. Для
каждого $k = 1, \dots, n$ ограничение~$Q$ на
$U_k = \spn(\vect{e}_1, \dots, \vect{e}_k) \subseteq V$ также
положительно определено (то же самое условие проверяется на
подмножестве ненулевых векторов). Матрица $Q|_{U_k}$ в базисе
$\{\vect{e}_1, \dots, \vect{e}_k\}$~--- в точности верхний левый блок
$k \times k$ матрицы~$\mat{A}$, и его определитель равен~$M_k$. По
\cref{thm:spectral} симметричная матрица $Q|_{U_k}$ имеет только
вещественные собственные значения, а положительная определённость
вынуждает каждое из них быть положительным (для собственного
вектора~$\vect{v}$ с собственным значением~$\mu$
$Q|_{U_k}(\vect{v}) = \mu \|\vect{v}\|^2$). Значит,~$M_k$~--- произведение
этих собственных значений~--- положителен.

\emph{(а)~$\Leftarrow$.} Пусть все $M_k > 0$ (полагаем $M_0 := 1$).
Покажем индукцией по~$n = \dim V$, что алгоритм Лагранжа в базисе
$\{\vect{e}_1, \dots, \vect{e}_n\}$ остаётся в Случае~1 на каждом шаге
и порождает диагональные коэффициенты
$\mu_1 = M_1$ и $\mu_k = M_k / M_{k-1}$ при $2 \le k \le n$.

\emph{Ключевое наблюдение.} Замена переменных в Случае~1 с ведущим
коэффициентом~$a_{11} \neq 0$ задаётся \emph{верхнетреугольной}
матрицей перехода~$\mat{S}$ с единицами на диагонали. Тогда верхний
левый $k \times k$-блок матрицы $\mat{S}\T \mat{A}\, \mat{S}$ есть
ровно $\mat{S}_k\T \mat{A}_k\, \mat{S}_k$, где~$\mat{S}_k$ и $\mat{A}_k$~---
верхние левые $k \times k$-блоки (это вытекает из блочной
верхнетреугольности~$\mat{S}$ с единичной диагональю), а
$\det(\mat{S}_k\T \mat{A}_k\, \mat{S}_k) = (\det \mat{S}_k)^2 \det \mat{A}_k = \det \mat{A}_k$.
Значит, ведущие угловые миноры формы~$Q$ при такой замене сохраняются.

\emph{База $n = 1$.} Тогда $Q = M_1 y_1^2$ с $\mu_1 = M_1 > 0$.

\emph{Шаг $n - 1 \Rightarrow n$.} Так как $a_{11} = M_1 > 0$, применим
Случай~1 и приведём~$Q$ к виду $M_1 y_1^2 + Q'(y_2, \dots, y_n)$ с
матрицей $\diag(M_1) \oplus \mat{B}_1$, где~$\mat{B}_1 \in \R^{(n-1) \times (n-1)}$~---
матрица формы~$Q'$. По ключевому наблюдению ведущие угловые миноры
сохранились: $M_k\bigl(\diag(M_1) \oplus \mat{B}_1\bigr) = M_k$.
Прямое вычисление того же минора блочно-диагональной матрицы даёт
$M_k\bigl(\diag(M_1) \oplus \mat{B}_1\bigr) = M_1 \cdot M_{k-1}(\mat{B}_1)$,
откуда
\[
  M_{k-1}(\mat{B}_1) \;=\; M_k / M_1 \;>\; 0
  \qquad (k = 1, \dots, n).
\]
Все ведущие угловые миноры~$\mat{B}_1$ положительны, так что по
предположению индукции дальнейший ход метода Лагранжа применённый
к~$Q'$ остаётся в Случае~1 и даёт коэффициенты
\[
  \mu_2 = M_1(\mat{B}_1) = M_2 / M_1,
  \qquad
  \mu_k = \frac{M_{k-1}(\mat{B}_1)}{M_{k-2}(\mat{B}_1)} = \frac{M_k/M_1}{M_{k-1}/M_1} = \frac{M_k}{M_{k-1}}
  \quad (3 \le k \le n).
\]

Итак, $Q$ конгруэнтна $\diag(M_1, M_2/M_1, \dots, M_n/M_{n-1})$ со
строго положительными диагональными элементами, и по
\cref{thm:inertia} форма~$Q$ положительно определена.
\end{proof}

\subsection*{Примеры}

\begin{example}[Приведение положительно определённой формы методом Лагранжа]
\label{ex:lagrange-pos-def}
Приведём $Q(x, y) = 2x^2 + 4xy + 5y^2$ к каноническому виду. Выделим
полный квадрат по~$x$:
\[
  Q = 2\bigl(x^2 + 2xy\bigr) + 5y^2 = 2(x + y)^2 - 2y^2 + 5y^2 = 2(x + y)^2 + 3y^2.
\]
Полагая $u = x + y$, $v = y$, получаем $Q = 2u^2 + 3v^2$ с обоими
положительными коэффициентами. Значит,~$Q$ положительно определена.

Проверка по Сильвестру: матрица $\mat{A} = \begin{pmatrix} 2 & 2 \\ 2 & 5 \end{pmatrix}$
имеет $M_1 = 2 > 0$ и $M_2 = 10 - 4 = 6 > 0$, что согласуется с
\cref{thm:sylvester}\,(a).
\end{example}

\begin{example}[Неопределённая форма]
\label{ex:lagrange-indefinite}
Для $Q(x, y) = x^2 - 4xy + 3y^2$ выделим полный квадрат по~$x$:
\[
  Q = (x - 2y)^2 - 4y^2 + 3y^2 = (x - 2y)^2 - y^2.
\]
Полагая $u = x - 2y$, $v = y$, получаем $Q = u^2 - v^2$, сигнатура
$(n_+, n_-) = (1, 1)$. Значит,~$Q$ \emph{неопределена}: $Q(1, 0) = 1 > 0$
и $Q(2, 1) = -1 < 0$.

По Сильвестру: $M_1 = 1 > 0$ и $M_2 = 3 - 4 = -1 < 0$, поэтому ни (а),
ни (б) из \cref{thm:sylvester} не выполняются~--- $Q$ не является
определённой ни одного знака.
\end{example}

\begin{example}[Чистый внедиагональный случай: $Q = 2xy$]
\label{ex:lagrange-off-diagonal}
Для $Q(x, y) = 2xy$ диагональные коэффициенты обнуляются, так что мы
попадаем в Случай~2 алгоритма Лагранжа. Замена $x = u + v$,
$y = u - v$ даёт
\[
  Q = 2(u + v)(u - v) = 2u^2 - 2v^2.
\]
Сигнатура $(1, 1)$, форма неопределена. По Сильвестру: $M_1 = 0$,
$M_2 = -1$. Условие~(а) \cref{thm:sylvester} нарушается уже на первом
шаге ($M_1 \not> 0$), и (б) тоже нарушается на $M_1$ ($M_1 \not< 0$);
форма не является определённой ни одного знака.
\end{example}

\begin{problem}
\label{prob:sylvester-parameter}
При каких значениях $\lambda \in \R$ форма
\[
  Q(x, y, z) \;=\; x^2 + 2y^2 + \lambda z^2 + 2xy + 2yz
\]
положительно определена?
\end{problem}

\begin{proof}[Решение]
Матрица формы~$Q$ имеет вид
\[
  \mat{A} = \begin{pmatrix} 1 & 1 & 0 \\ 1 & 2 & 1 \\ 0 & 1 & \lambda \end{pmatrix},
\]
её угловые миноры суть $M_1 = 1$, $M_2 = 2 - 1 = 1$ и (разлагая
определитель по третьей строке)
\[
  M_3 = \det \mat{A} = 0 - 1 \cdot (1 \cdot 1 - 0 \cdot 1) + \lambda \cdot (1 \cdot 2 - 1 \cdot 1)
      = -1 + \lambda = \lambda - 1.
\]
По \cref{thm:sylvester}\,(a) форма~$Q$ положительно определена тогда и
только тогда, когда $M_1, M_2, M_3 > 0$, то есть при $\lambda > 1$.

Контроль на границе $\lambda = 1$: $M_3 = 0$ означает, что~$\mat{A}$
вырождена, поэтому~$Q$ положительно полуопределена, но не положительно
определена ($Q$ обращается в нуль на ненулевом векторе из ядра; в
самом деле, $Q(1, -1, 1) = 1 + 2 + 1 - 2 - 2 = 0$).
\end{proof}
