% Topic 1.1.09 — Differentiation. Taylor's Formula.

\section{Дифференцирование. Формула Тейлора}
\label{sec:1.1.09-derivatives-taylor}

\begin{ticket}
Дифференцирование функций. Применение производной для нахождения экстремумов функций. Формула Тейлора.
\end{ticket}

\subsection*{Теория}

Рассматриваем вещественнозначные функции одной вещественной переменной,
определённые на открытом подмножестве~$\R$. Через $x_0$ всюду
обозначаем выбранную внутреннюю точку области определения.

\begin{definition}
\label{def:derivative}
Функция~$f$ называется \emph{дифференцируемой в точке~$x_0$}, если
существует и конечен предел
\[
  f'(x_0) \;=\; \lim_{h \to 0} \frac{f(x_0 + h) - f(x_0)}{h}.
\]
Эквивалентно, существует $A \in \R$, для которого
\[
  f(x_0 + h) \;=\; f(x_0) + A \cdot h + o(h)
  \qquad \text{при } h \to 0,
\]
причём в этом случае $A = f'(x_0)$. Функция~$f$ \emph{дифференцируема
на открытом множестве~$U$}, если она дифференцируема в каждой точке
$x_0 \in U$; возникающая при этом функция $x \mapsto f'(x)$ называется
\emph{производной}.
\end{definition}

Стандартные правила дифференцирования~--- линейность, произведение,
частное, цепное~--- считаем известными; они получаются прямо из
\cref{def:derivative} преобразованием разностных отношений.
Сосредоточимся на том, что говорит производная о самой функции.

\begin{proposition}[Из дифференцируемости следует непрерывность]
\label{prop:diff-implies-cont}
Если~$f$ дифференцируема в точке~$x_0$, то~$f$ непрерывна в~$x_0$.
\end{proposition}
\proofof{diff-implies-cont}

\begin{definition}
\label{def:local-extremum}
Функция~$f$ имеет в точке~$x_0$ \emph{локальный максимум} (соответственно
\emph{локальный минимум}), если существует окрестность $U \ni x_0$, в
которой $f(x) \le f(x_0)$ (соответственно $f(x) \ge f(x_0)$) при всех
$x \in U$. \emph{Локальным экстремумом} называется локальный максимум
или локальный минимум; экстремум \emph{строгий}, если равенство
достигается лишь при $x = x_0$.
\end{definition}

\begin{theorem}[Ферма]
\label{thm:fermat}
Если~$f$ имеет локальный экстремум во внутренней точке~$x_0$ и
дифференцируема в~$x_0$, то $f'(x_0) = 0$.
\end{theorem}
\proofof{fermat}

\begin{theorem}[Ролль]
\label{thm:rolle}
Пусть~$f$ непрерывна на $[a, b]$, дифференцируема на $(a, b)$ и
$f(a) = f(b)$. Тогда найдётся $c \in (a, b)$ с $f'(c) = 0$.
\end{theorem}
\proofof{rolle}

\begin{theorem}[Лагранж: теорема о среднем]
\label{thm:lagrange-mvt}
Пусть~$f$ непрерывна на $[a, b]$ и дифференцируема на $(a, b)$. Тогда
найдётся $c \in (a, b)$, для которого
\[
  f(b) - f(a) \;=\; f'(c)\,(b - a).
\]
\end{theorem}
\proofof{lagrange-mvt}

\begin{theorem}[Признак экстремума по первой производной]
\label{thm:first-deriv-test}
Пусть~$f$ непрерывна в окрестности точки~$x_0$ и дифференцируема в
проколотой окрестности. Если $f'(x) > 0$ при $x < x_0$, близких
к~$x_0$, и $f'(x) < 0$ при $x > x_0$, близких к~$x_0$, то~$f$ имеет в
точке~$x_0$ строгий локальный максимум. Зеркальное распределение
знаков даёт строгий локальный минимум.
\end{theorem}
\proofof{first-deriv-test}

\begin{theorem}[Формула Тейлора с остатком в форме Пеано]
\label{thm:taylor-peano}
Пусть~$f$ имеет в точке~$x_0$ производные до порядка~$n$
включительно. Тогда при $x \to x_0$
\[
  f(x) \;=\; \sum_{k=0}^{n} \frac{f^{(k)}(x_0)}{k!}\,(x - x_0)^k \;+\; o\bigl((x - x_0)^n\bigr).
\]
\end{theorem}
\proofof{taylor-peano}

\begin{theorem}[Формула Тейлора с остатком в форме Лагранжа]
\label{thm:taylor-lagrange}
Пусть~$f$ имеет производные до порядка $n+1$ включительно на открытом
интервале, содержащем точки $x_0$ и~$x$. Тогда найдётся~$c$ строго
между $x_0$ и~$x$, для которого
\[
  f(x) \;=\; \sum_{k=0}^{n} \frac{f^{(k)}(x_0)}{k!}\,(x - x_0)^k
  \;+\; \frac{f^{(n+1)}(c)}{(n+1)!}\,(x - x_0)^{n+1}.
\]
\end{theorem}
\proofof{taylor-lagrange}

\begin{corollary}[Признак экстремума по второй производной]
\label{cor:second-deriv-test}
Пусть~$f$ дважды дифференцируема в точке~$x_0$ и $f'(x_0) = 0$.
\begin{itemize}
  \item Если $f''(x_0) > 0$, то~$f$ имеет в~$x_0$ строгий локальный минимум.
  \item Если $f''(x_0) < 0$, то~$f$ имеет в~$x_0$ строгий локальный максимум.
  \item Если $f''(x_0) = 0$, признак не даёт ответа.
\end{itemize}
\end{corollary}
\proofof{second-deriv-test}

\begin{remark}
Доказательства ниже опираются лишь на \cref{thm:rolle,thm:lagrange-mvt}
и обобщённую теорему Коши о среднем (по существу теорема Ролля,
применённая ко вспомогательной двухкомпонентной функции). Полное
изложение, включая интегральную форму остатка и следствия для
вещественно-аналитических функций, см.\ в~\cite{Zorich1981}.
\end{remark}

\subsection*{Доказательства}

\begin{proof}[\Cref{prop:diff-implies-cont}]
\proofanchor{diff-implies-cont}
При $h \neq 0$
\[
  f(x_0 + h) - f(x_0) \;=\; \frac{f(x_0 + h) - f(x_0)}{h} \cdot h.
\]
Правая часть стремится к $f'(x_0) \cdot 0 = 0$ при $h \to 0$, поэтому
$f(x_0 + h) \to f(x_0)$~--- это и есть непрерывность в~$x_0$.
\end{proof}

\begin{proof}[\Cref{thm:fermat}]
\proofanchor{fermat}
Пусть~$x_0$~--- локальный максимум (случай минимума зеркален). При
всех достаточно малых~$h$ имеем $f(x_0 + h) \le f(x_0)$, то есть
$f(x_0 + h) - f(x_0) \le 0$. Поэтому
\[
  \frac{f(x_0 + h) - f(x_0)}{h} \;\le\; 0 \text{ при } h > 0,
  \qquad
  \frac{f(x_0 + h) - f(x_0)}{h} \;\ge\; 0 \text{ при } h < 0.
\]
Оба односторонних предела существуют и равны $f'(x_0)$ в силу
дифференцируемости, поэтому одновременно $f'(x_0) \le 0$ и
$f'(x_0) \ge 0$, откуда $f'(x_0) = 0$.
\end{proof}

\begin{proof}[\Cref{thm:rolle}]
\proofanchor{rolle}
По непрерывности на отрезке $[a, b]$ функция~$f$ достигает максимума и
минимума в некоторых $c_M, c_m \in [a, b]$. Если оба эти значения
находятся на концах, то $f(a) = f(b)$ принуждает максимум совпадать с
минимумом, то есть~$f$ постоянна; тогда $f'(c) = 0$ при любом
$c \in (a, b)$. Иначе хотя бы одна из точек $c_M, c_m$ лежит внутри
$(a, b)$, и применение \cref{thm:fermat} в этой точке даёт $f'(c) = 0$.
\end{proof}

\begin{proof}[\Cref{thm:lagrange-mvt}]
\proofanchor{lagrange-mvt}
Применим \cref{thm:rolle} ко вспомогательной функции
\[
  g(x) \;=\; f(x) - f(a) - \frac{f(b) - f(a)}{b - a}\,(x - a),
\]
для которой $g(a) = 0 = g(b)$, $g$ непрерывна на $[a, b]$ и
дифференцируема на $(a, b)$. Теорема Ролля даёт $c \in (a, b)$ с
$g'(c) = 0$, то есть $f'(c) = (f(b) - f(a))/(b - a)$.
\end{proof}

\begin{proof}[\Cref{thm:first-deriv-test}]
\proofanchor{first-deriv-test}
Рассмотрим случай $f' > 0$ слева и $f' < 0$ справа от~$x_0$
(симметричное распределение знаков отрабатывается зеркально). Пусть
$\delta > 0$ таково, что~$f$ непрерывна на $(x_0 - \delta, x_0 + \delta)$,
дифференцируема на $(x_0 - \delta, x_0) \cup (x_0, x_0 + \delta)$, причём
$f' > 0$ слева и $f' < 0$ справа.

Для $x \in (x_0 - \delta, x_0)$ применим теорему Лагранжа
(\cref{thm:lagrange-mvt}) к~$f$ на отрезке $[x, x_0]$: найдётся
$\xi \in (x, x_0)$ с $f(x_0) - f(x) = f'(\xi)(x_0 - x)$. Здесь
$f'(\xi) > 0$ (так как $\xi$ слева от~$x_0$) и $x_0 - x > 0$, поэтому
$f(x_0) > f(x)$.

Для $x \in (x_0, x_0 + \delta)$ аналогично на $[x_0, x]$ получаем
$\eta \in (x_0, x)$ с $f(x) - f(x_0) = f'(\eta)(x - x_0)$. Здесь
$f'(\eta) < 0$ и $x - x_0 > 0$, поэтому $f(x) < f(x_0)$.

В обоих случаях $f(x_0) > f(x)$ при $x \neq x_0$, $|x - x_0| < \delta$,~---
строгий локальный максимум.
\end{proof}

\begin{proof}[\Cref{thm:taylor-peano}]
\proofanchor{taylor-peano}
Пусть $T_n(x) = \sum_{k=0}^{n} \frac{f^{(k)}(x_0)}{k!} (x - x_0)^k$~---
многочлен Тейлора, $r_n(x) = f(x) - T_n(x)$~--- остаток. Покажем, что
$r_n(x) = o((x - x_0)^n)$ при $x \to x_0$.

По построению $r_n(x_0) = 0$ и $r_n^{(k)}(x_0) = 0$ при
$k = 0, 1, \dots, n$ ($k$-е производные многочлена~$T_n$ и функции~$f$
совпадают в~$x_0$ для всех $k$ до~$n$).

Применим обобщённую теорему Коши о среднем итеративно. Для каждого~$x$
вблизи~$x_0$ найдётся точка~$\xi_1$ строго между $x_0$ и~$x$, для
которой
\[
  \frac{r_n(x)}{(x - x_0)^n}
  \;=\; \frac{r_n(x) - r_n(x_0)}{(x - x_0)^n - 0}
  \;=\; \frac{r_n'(\xi_1)}{n\,(\xi_1 - x_0)^{n-1}}.
\]
Применяя ту же теорему ещё раз, к $r_n'$ и $(\,\cdot\, - x_0)^{n-1}$
на отрезке между $x_0$ и~$\xi_1$, и используя $r_n'(x_0) = 0$,
получаем точку~$\xi_2$ строго между $x_0$ и~$\xi_1$, для которой
\[
  \frac{r_n'(\xi_1)}{n\,(\xi_1 - x_0)^{n-1}}
  \;=\; \frac{r_n''(\xi_2)}{n(n-1)\,(\xi_2 - x_0)^{n-2}}.
\]
Продолжая до $n - 1$ итераций, получаем точку~$\xi_{n-1}$ строго между
$x_0$ и~$x$, для которой
\[
  \frac{r_n(x)}{(x - x_0)^n}
  \;=\; \frac{r_n^{(n-1)}(\xi_{n-1})}{n!\,(\xi_{n-1} - x_0)}.
\]
При $x \to x_0$ все промежуточные точки $\xi_1, \dots, \xi_{n-1}$
стремятся к~$x_0$, и правая часть, по самому определению $n$-й
производной (примененному к функции $r_n^{(n-1)}$),
\[
  \lim_{\xi \to x_0} \frac{r_n^{(n-1)}(\xi) - r_n^{(n-1)}(x_0)}{n!\,(\xi - x_0)}
  \;=\; \frac{r_n^{(n)}(x_0)}{n!} \;=\; 0,
\]
так как $r_n^{(n-1)}(x_0) = 0$ и $r_n^{(n)}(x_0) = 0$.
Следовательно, $r_n(x)/(x - x_0)^n \to 0$, то есть
$r_n(x) = o((x - x_0)^n)$.
\end{proof}

\begin{proof}[\Cref{thm:taylor-lagrange}]
\proofanchor{taylor-lagrange}
Зафиксируем $x \neq x_0$ и определим на отрезке между $x_0$ и~$x$
функции
\[
  g(t) \;=\; f(x) - \sum_{k=0}^{n} \frac{f^{(k)}(t)}{k!}\,(x - t)^k,
  \qquad
  h(t) \;=\; (x - t)^{n+1}.
\]
Обе функции $g$ и $h$ дифференцируемы на этом отрезке; вычисление
$g'$ телескопично (все слагаемые, кроме последнего, сокращаются) и
даёт
\[
  g'(t) \;=\; -\frac{f^{(n+1)}(t)}{n!}\,(x - t)^n,
  \qquad
  h'(t) \;=\; -(n+1)(x - t)^n.
\]
В концах $t = x_0$ и $t = x$:
\[
  g(x_0) = f(x) - \sum_{k=0}^{n} \frac{f^{(k)}(x_0)}{k!}\,(x - x_0)^k,
  \qquad
  g(x) = 0,
  \qquad
  h(x_0) = (x - x_0)^{n+1},
  \qquad
  h(x) = 0.
\]
Обобщённая теорема Коши о среднем, применённая к паре $(g, h)$, даёт
точку~$c$ строго между $x_0$ и~$x$, для которой
\[
  \frac{g(x_0) - g(x)}{h(x_0) - h(x)} \;=\; \frac{g'(c)}{h'(c)}
  \;=\; \frac{-f^{(n+1)}(c)\,(x - c)^n / n!}{-(n+1)(x - c)^n}
  \;=\; \frac{f^{(n+1)}(c)}{(n+1)!}.
\]
Эквивалентно, $g(x_0) = \frac{f^{(n+1)}(c)}{(n+1)!}\,(x - x_0)^{n+1}$,
что и есть требуемая формула Тейлора.
\end{proof}

\begin{proof}[\Cref{cor:second-deriv-test}]
\proofanchor{second-deriv-test}
Применим формулу Тейлора с остатком в форме Пеано порядка~$2$
в~$x_0$, учитывая $f'(x_0) = 0$:
\[
  f(x) - f(x_0) \;=\; \tfrac{1}{2} f''(x_0)\,(x - x_0)^2 + o\bigl((x - x_0)^2\bigr).
\]
Если $f''(x_0) > 0$, главный член $\tfrac{1}{2}f''(x_0)(x - x_0)^2$
строго положителен при $x \neq x_0$, близких к~$x_0$, и доминирует
над $o((x - x_0)^2)$, поэтому $f(x) > f(x_0)$ в проколотой
окрестности~--- это строгий локальный минимум. Симметричное
рассуждение даёт случай максимума. При $f''(x_0) = 0$ возможны оба
знака ($f(x) = x^4$ имеет минимум, $f(x) = -x^4$~--- максимум,
$f(x) = x^3$~--- ни того, ни другого, причём у всех первая и вторая
производные в~$0$ обращаются в нуль).
\end{proof}

\subsection*{Примеры}

\begin{example}[Экстремумы кубической функции]
\label{ex:cubic-extrema}
Пусть $f(x) = x^3 - 3x^2$. Тогда $f'(x) = 3x^2 - 6x = 3x(x - 2)$, и
критические точки (где $f'$ обращается в нуль)~--- $x = 0$ и $x = 2$.
Имеем $f''(x) = 6x - 6$, откуда $f''(0) = -6 < 0$ и $f''(2) = 6 > 0$.
По \cref{cor:second-deriv-test} функция~$f$ имеет в точке $x = 0$
строгий локальный максимум ($f(0) = 0$) и в точке $x = 2$~--- строгий
локальный минимум ($f(2) = -4$).
\end{example}

\begin{example}[Разложение Тейлора для $e^x$]
\label{ex:taylor-exp}
Возьмём $f(x) = e^x$. Все производные равны $f^{(k)}(x) = e^x$, а
значит, $f^{(k)}(0) = 1$. Формула Тейлора с остатком в форме
Лагранжа (\cref{thm:taylor-lagrange}) при $x_0 = 0$ даёт для всякого
$x \in \R$ и каждого~$n$
\[
  e^x \;=\; \sum_{k=0}^{n} \frac{x^k}{k!} \;+\; \frac{e^c}{(n+1)!}\,x^{n+1}
\]
при некотором $c$ строго между $0$ и~$x$. Оценивая $e^c \le e^{|x|}$:
\[
  \Bigl| e^x - \sum_{k=0}^{n} \frac{x^k}{k!} \Bigr|
  \;\le\; \frac{e^{|x|} \,|x|^{n+1}}{(n+1)!}.
\]
Правая часть стремится к нулю при $n \to \infty$ для любого
фиксированного~$x$, что и даёт привычный ряд
$e^x = \sum_{k=0}^{\infty} x^k / k!$.
\end{example}

\begin{example}[Тейлор для пределов]
\label{ex:taylor-limit}
Вычислим $\lim_{x \to 0} (\sin x - x) / x^3$. Разложение с остатком в
форме Пеано $\sin x = x - x^3/6 + o(x^3)$ даёт
\[
  \frac{\sin x - x}{x^3} \;=\; \frac{-x^3/6 + o(x^3)}{x^3}
  \;=\; -\tfrac{1}{6} + o(1) \;\to\; -\tfrac{1}{6}.
\]
Тот же предел можно получить трёхкратным применением правила
Лопиталя, но разложение Тейлора сводит вычисление к одному шагу.
\end{example}

\begin{problem}
\label{prob:taylor-estimate-e}
По многочлену Тейлора функции $e^x$ в точке $x_0 = 0$ вычислите
$e \approx \sum_{k=0}^{n} 1/k!$ при некотором~$n$ и подберите такое~$n$,
которое гарантирует точность четырёх десятичных знаков:
$|e - \sum_{k=0}^{n} 1/k!| < 5 \times 10^{-5}$.
\end{problem}

\begin{proof}[Решение]
По \cref{ex:taylor-exp} при $x = 1$
\[
  \Bigl| e - \sum_{k=0}^{n} \frac{1}{k!} \Bigr| \;\le\; \frac{e^1}{(n+1)!}
  \;\le\; \frac{3}{(n+1)!},
\]
где использована грубая оценка $e \le 3$. Требуется
$(n+1)! \ge 3 / (5 \times 10^{-5}) = 60000$. Вычисляем факториалы:
$7! = 5040$, $8! = 40320$, $9! = 362880$. Так что $(n+1)! \ge 60000$
впервые достигается при $n + 1 = 9$, то есть при $n = 8$.

Получаем оценку
$e \approx \sum_{k=0}^{8} 1/k! = 1 + 1 + 1/2 + 1/6 + 1/24 + 1/120 + 1/720 + 1/5040 + 1/40320 \approx 2{,}71828$,
что согласуется с $e = 2{,}71828\ldots$ во всех пяти показанных
десятичных знаках. Оценка $3/9! \approx 8{,}3 \times 10^{-6}$
заведомо не превосходит требуемого допуска.
\end{proof}
