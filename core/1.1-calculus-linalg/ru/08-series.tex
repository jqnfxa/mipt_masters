\section{Числовые и функциональные ряды}
\label{sec:1.1.08-series}

\begin{ticket}
Ряды. Числовые и функциональные ряды. Признаки сходимости (Даламбера, Коши, интегральный, Лейбница). Абсолютно и условно сходящиеся ряды.
\end{ticket}

\subsection*{Теория}

В этом разделе работаем с вещественными числами; утверждения об
абсолютной сходимости дословно переносятся на комплексные ряды через
$|z| = \sqrt{z\bar{z}}$.

\begin{definition}
\label{def:series-convergence}
\emph{Числовым рядом} называется формальная сумма
$\sum_{n=1}^{\infty} a_n$, отвечающая последовательности $(a_n)$
вещественных чисел. Её \emph{частичные суммы}~---
$S_N = \sum_{n=1}^{N} a_n$. Ряд \emph{сходится} (к \emph{сумме}~$S$),
если $\lim_{N \to \infty} S_N = S$, и \emph{расходится} в противном
случае.
\end{definition}

\begin{proposition}[Необходимое условие сходимости]
\label{prop:necessary}
Если $\sum a_n$ сходится, то $a_n \to 0$ при $n \to \infty$.
\end{proposition}
\proofof{necessary}

\begin{theorem}[Критерий Коши]
\label{thm:series-cauchy}
Ряд $\sum a_n$ сходится тогда и только тогда, когда для любого
$\varepsilon > 0$ найдётся такое~$N$, что
$\bigl|\sum_{n=p+1}^{q} a_n\bigr| < \varepsilon$ при всех $q > p \ge N$.
\end{theorem}

Это просто критерий Коши для последовательности~$(S_N)$ в полном
пространстве~$\R$; будем пользоваться им без отдельного доказательства.

\begin{definition}
\label{def:abs-cond-convergence}
Ряд $\sum a_n$ \emph{сходится абсолютно}, если сходится $\sum |a_n|$.
Сходящийся ряд, не сходящийся абсолютно, называется \emph{сходящимся
условно}.
\end{definition}

\begin{theorem}[Из абсолютной сходимости следует сходимость]
\label{thm:abs-implies-conv}
Если $\sum |a_n|$ сходится, то и $\sum a_n$ сходится, причём
$\bigl|\sum a_n\bigr| \le \sum |a_n|$.
\end{theorem}
\proofof{abs-implies-conv}

\medskip

\noindent\textbf{Признаки сходимости рядов с неотрицательными
членами.}

\begin{theorem}[Признак Даламбера]
\label{thm:dalembert}
Пусть $a_n \neq 0$ при всех~$n$ и существует предел
$q = \lim_{n \to \infty} \bigl|\tfrac{a_{n+1}}{a_n}\bigr|$ в
$[0, +\infty]$. Тогда
\begin{enumerate}[label=(\alph*)]
  \item при $q < 1$ ряд $\sum a_n$ сходится абсолютно;
  \item при $q > 1$ (включая $q = +\infty$) ряд расходится
        (более того, $a_n \not\to 0$);
  \item при $q = 1$ признак не даёт ответа.
\end{enumerate}
\end{theorem}
\proofof{dalembert}

\begin{theorem}[Радикальный признак Коши]
\label{thm:cauchy-root}
Пусть $q = \limsup_{n \to \infty} \sqrt[n]{|a_n|} \in [0, +\infty]$.
\begin{enumerate}[label=(\alph*)]
  \item при $q < 1$ ряд $\sum a_n$ сходится абсолютно;
  \item при $q > 1$ ряд $\sum a_n$ расходится;
  \item при $q = 1$ признак не даёт ответа.
\end{enumerate}
\end{theorem}
\proofof{cauchy-root}

\begin{theorem}[Интегральный признак Коши--Маклорена]
\label{thm:integral-test}
Пусть $f \colon [1, +\infty) \to [0, +\infty)$~--- невозрастающая
неотрицательная функция и $a_n = f(n)$. Тогда ряд
$\sum_{n=1}^{\infty} a_n$ и несобственный интеграл
$\int_1^{+\infty} f(x)\, dx$ сходятся либо расходятся одновременно;
кроме того,
\[
  \int_{1}^{N+1} f(x)\, dx \;\le\; \sum_{n=1}^{N} a_n \;\le\; a_1 + \int_{1}^{N} f(x)\, dx.
\]
\end{theorem}
\proofof{integral-test}

\medskip

\noindent\textbf{Знакочередующиеся ряды.}

\begin{theorem}[Признак Лейбница]
\label{thm:leibniz}
Пусть $(a_n)$~--- невозрастающая последовательность неотрицательных
чисел с $a_n \to 0$. Тогда знакочередующийся ряд
$\sum_{n=1}^{\infty} (-1)^{n+1} a_n$ сходится, и погрешность остатка
удовлетворяет неравенству
\[
  |S - S_N| \;\le\; a_{N+1}.
\]
\end{theorem}
\proofof{leibniz}

\medskip

\noindent\textbf{Функциональные ряды и равномерная сходимость.}

\begin{definition}
\label{def:functional-series}
Пусть $D \subseteq \R$ и $f_n \colon D \to \R$ при $n = 1, 2, \dots$.
\emph{Функциональный ряд} $\sum f_n$ \emph{сходится поточечно на $D$},
если $\sum f_n(x)$ сходится при каждом $x \in D$, и \emph{сходится
равномерно на $D$}, если
$\sup_{x \in D} \bigl|S(x) - S_N(x)\bigr| \to 0$ при $N \to \infty$,
где $S_N(x) = \sum_{n=1}^{N} f_n(x)$, а $S(x)$~--- поточечная сумма.
\end{definition}

\begin{theorem}[Признак Вейерштрасса]
\label{thm:weierstrass-m}
Если найдутся такие константы $M_n \ge 0$, что $|f_n(x)| \le M_n$ при
всех $x \in D$ и всех~$n$, и при этом $\sum M_n$ сходится, то
$\sum f_n$ сходится равномерно (и абсолютно) на~$D$.
\end{theorem}
\proofof{weierstrass-m}

\begin{remark}
Равномерная сходимость~--- адекватное понятие для перестановки ряда с
пределами, производными и интегралами (непрерывность суммы, почленное
интегрирование и т.\,п.). Эти приложения встретятся в
\cref{sec:1.1.12-fourier-series,sec:1.1.13-fourier-transform}.
Подробное обсуждение всех результатов раздела см.\ в~\cite{Zorich1981}.
\end{remark}

\subsection*{Доказательства}

\begin{proof}[\Cref{prop:necessary}]
\proofanchor{necessary}
Если $S_N \to S$, то $a_N = S_N - S_{N-1} \to S - S = 0$.
\end{proof}

\begin{proof}[\Cref{thm:abs-implies-conv}]
\proofanchor{abs-implies-conv}
Применим критерий Коши (\cref{thm:series-cauchy}) к обоим рядам.
При $q > p$ неравенство треугольника даёт
\[
  \Bigl| \sum_{n=p+1}^{q} a_n \Bigr| \;\le\; \sum_{n=p+1}^{q} |a_n|.
\]
Если $\sum |a_n|$ сходится, то правая часть меньше~$\varepsilon$
при $p, q \ge N$, а тогда и левая, так что $\sum a_n$ сходится.
Переходя в неравенстве
$|\sum_{n=1}^{q} a_n| \le \sum_{n=1}^{q} |a_n|$ к пределу
$q \to \infty$, получаем нужную оценку для самих сумм.
\end{proof}

\begin{proof}[\Cref{thm:dalembert}]
\proofanchor{dalembert}
\emph{(а) $q < 1$.} Возьмём $q'$ с $q < q' < 1$. По определению
предела найдётся такое~$N$, что
$\bigl|\tfrac{a_{n+1}}{a_n}\bigr| \le q'$ при $n \ge N$. По индукции
$|a_{N+k}| \le q'^k |a_N|$, так что $\sum_{n \ge N} |a_n|$
мажорируется сходящейся геометрической прогрессией
$|a_N| \sum_k q'^k = |a_N|/(1 - q')$. Значит, $\sum |a_n|$ сходится.

\emph{(б) $q > 1$.} Возьмём $q'$ с $1 < q' < q$. Найдётся такое~$N$,
что $|a_{n+1}/a_n| \ge q'$ при $n \ge N$, откуда
$|a_n| \ge q'^{n - N}|a_N|$ при $n \ge N$. Поскольку $q' > 1$, эта
оценка снизу неограниченно возрастает, поэтому $a_n \not\to 0$ и ряд
расходится по \cref{prop:necessary}.

\emph{(в) $q = 1$.} Гармонический ряд $\sum 1/n$ имеет $q = 1$ и
расходится, а ряд $\sum 1/n^2$ имеет $q = 1$ и сходится (см.\
примеры), так что общего вывода сделать нельзя.
\end{proof}

\begin{proof}[\Cref{thm:cauchy-root}]
\proofanchor{cauchy-root}
\emph{(а) $q < 1$.} Возьмём $q'$ с $q < q' < 1$. По определению
верхнего предела найдётся такое~$N$, что $\sqrt[n]{|a_n|} \le q'$
при $n \ge N$, то есть $|a_n| \le q'^n$. Геометрический ряд
$\sum q'^n$ сходится, поэтому $\sum |a_n|$ сходится по признаку
сравнения.

\emph{(б) $q > 1$.} Существует бесконечно много~$n$, для которых
$\sqrt[n]{|a_n|} > 1$, то есть $|a_n| > 1$. Значит, $a_n \not\to 0$,
и ряд расходится по \cref{prop:necessary}.

\emph{(в) $q = 1$.} Те же примеры, что и для Даламбера, показывают
неоднозначность.
\end{proof}

\begin{proof}[\Cref{thm:integral-test}]
\proofanchor{integral-test}
В силу невозрастания~$f$ для каждого целого $n \ge 1$ и
$x \in [n, n+1]$ имеем $f(n+1) \le f(x) \le f(n)$. Интегрируя:
\[
  a_{n+1} = f(n+1) \;\le\; \int_n^{n+1} f(x)\, dx \;\le\; f(n) = a_n.
\]
Суммируя по $n = 1, \dots, N - 1$:
\[
  \sum_{n=2}^{N} a_n \;\le\; \int_1^{N} f(x)\, dx \;\le\; \sum_{n=1}^{N-1} a_n.
\]
Прибавляя $a_1$ к обеим частям левого неравенства, получаем
$\sum_{n=1}^{N} a_n \le a_1 + \int_1^N f$ — это правая половина
формулировки. Сдвигая в правом неравенстве индекс $N \to N + 1$:
$\int_1^{N+1} f \le \sum_{n=1}^{N} a_n$ — это левая половина.
Частичные суммы и частичные интегралы неубывают по~$N$ и оценивают
друг друга с точностью до константы, поэтому сходятся либо расходятся
одновременно.
\end{proof}

\begin{proof}[\Cref{thm:leibniz}]
\proofanchor{leibniz}
Положим $S_N = \sum_{n=1}^{N} (-1)^{n+1} a_n$. Чётные частичные
суммы удовлетворяют
\[
  S_{2k+2} - S_{2k} = a_{2k+1} - a_{2k+2} \ge 0,
\]
то есть $(S_{2k})$ неубывает. Нечётные частичные суммы:
\[
  S_{2k+1} - S_{2k-1} = -a_{2k} + a_{2k+1} \le 0,
\]
то есть $(S_{2k+1})$ невозрастает. Поскольку $S_{2k} \le S_{2k+1}$
(из $S_{2k+1} - S_{2k} = a_{2k+1} \ge 0$), всякая $S_{2k}$ ограничена
сверху $S_1 = a_1$, а всякая $S_{2k+1}$~--- снизу
$S_2 = a_1 - a_2 \ge 0$. Две монотонные ограниченные
последовательности сходятся; обозначим их пределы через $L$ и $L'$.
Из $S_{2k+1} - S_{2k} = a_{2k+1} \to 0$ получаем $L = L'$, и
$S_N \to L$.

Для оценки погрешности заметим, что при чётном $N$ имеем
$L \in [S_{N}, S_{N+1}]$, а при нечётном~--- $L \in [S_{N+1}, S_N]$;
в обоих случаях $|S - S_N| \le |S_{N+1} - S_N| = a_{N+1}$.
\end{proof}

\begin{proof}[\Cref{thm:weierstrass-m}]
\proofanchor{weierstrass-m}
Воспользуемся равномерным критерием Коши: ряд $\sum f_n$ сходится
равномерно тогда и только тогда, когда для любого $\varepsilon > 0$
найдётся такое~$N$, что
$\sup_{x \in D} \bigl|\sum_{n=p+1}^{q} f_n(x)\bigr| < \varepsilon$
при всех $q > p \ge N$. Оценим супремум через хвост ряда из $M_n$:
\[
  \Bigl| \sum_{n=p+1}^{q} f_n(x) \Bigr|
  \;\le\; \sum_{n=p+1}^{q} |f_n(x)|
  \;\le\; \sum_{n=p+1}^{q} M_n.
\]
Из сходимости $\sum M_n$ следует, что правая часть меньше
$\varepsilon$ равномерно по~$x$ при достаточно больших $p, q$.
\end{proof}

\subsection*{Примеры}

\begin{example}[Геометрический ряд]
\label{ex:geometric}
При $q \in \R$ геометрический ряд $\sum_{n=0}^{\infty} q^n$ сходится
тогда и только тогда, когда $|q| < 1$, и в этом случае его сумма
равна $1/(1 - q)$. Действительно,
$S_N = (1 - q^{N+1})/(1 - q) \to 1/(1 - q)$ при $|q| < 1$, а при
$|q| \ge 1$ имеем $|q|^{N+1} \not\to 0$. На всём интервале $|q| < 1$
ряд сходится абсолютно.
\end{example}

\begin{example}[Гармонический ряд и ряды $\sum 1/n^p$]
\label{ex:p-series}
При $p > 0$ применим интегральный признак к функции $f(x) = 1/x^p$:
$\int_1^{\infty} x^{-p}\, dx$ сходится тогда и только тогда, когда
$p > 1$. Значит, $\sum_{n=1}^{\infty} 1/n^p$ сходится тогда и только
тогда, когда $p > 1$. В частности, гармонический ряд $\sum 1/n$
($p = 1$) расходится, а $\sum 1/n^2$ сходится.

Признаки Даламбера и Коши на $p$-рядах не дают ответа
($\lim |a_{n+1}/a_n| = 1$ и $\lim \sqrt[n]{1/n^p} = 1$),~--- это
иллюстрация случая $q = 1$.
\end{example}

\begin{example}[$\sum 1/n!$ по признаку Даламбера]
\label{ex:factorial-series}
Для $a_n = 1/n!$ имеем
$\bigl|a_{n+1}/a_n\bigr| = 1/(n+1) \to 0 < 1$, поэтому по
\cref{thm:dalembert}\,(а) ряд сходится абсолютно. (Сумма равна
$e - 1$, но здесь нас интересует только сходимость.)
\end{example}

\begin{example}[Условная сходимость]
\label{ex:alt-harmonic}
Знакочередующийся гармонический ряд
$\sum_{n=1}^{\infty} (-1)^{n+1}/n$ имеет $a_n = 1/n \downarrow 0$
монотонно, поэтому сходится по \cref{thm:leibniz} (его сумма равна
$\ln 2$). Однако $\sum |a_n| = \sum 1/n$ расходится, так что ряд
сходится \emph{условно}, не абсолютно.
\end{example}

\begin{problem}
\label{prob:series-classify}
Для каждого $\alpha > 0$ установите, является ли ряд
\[
  \sum_{n=1}^{\infty} \frac{(-1)^n}{n^{\alpha}}
\]
абсолютно сходящимся, условно сходящимся или расходящимся.
\end{problem}

\begin{proof}[Решение]
Ряд из модулей~--- $p$-ряд $\sum 1/n^{\alpha}$, который по
\cref{ex:p-series} сходится тогда и только тогда, когда $\alpha > 1$.

\emph{$\alpha > 1$.} Имеет место абсолютная сходимость, а значит, и
сходимость (\cref{thm:abs-implies-conv}). Ряд сходится абсолютно.

\emph{$0 < \alpha \le 1$.} Абсолютной сходимости нет. Применим
признак Лейбница к $a_n = 1/n^{\alpha}$: последовательность
неотрицательна, монотонно убывает (поскольку $\alpha > 0$) и стремится
к~$0$. Значит, знакочередующийся ряд $\sum (-1)^n/n^{\alpha}$
сходится по \cref{thm:leibniz}. Поскольку абсолютная сходимость
отсутствует, сходимость \emph{условная}.

Итог: при $\alpha > 1$ ряд сходится абсолютно, при $0 < \alpha \le 1$~---
условно, и ни при каких $\alpha > 0$ не расходится.
\end{proof}
